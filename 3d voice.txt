    <script>
        const card = document.getElementById('card');
        const visualizer = document.getElementById('visualizer');
        const audioElement = document.getElementById('audio-element');
        const playBtn = document.getElementById('play-btn');
        const micBtn = document.getElementById('mic-btn');
        const audioUpload = document.getElementById('audio-upload');
        const trackName = document.getElementById('track-name');

        const BAR_COUNT = 16;
        const bars = [];

        // 1. Generate Bars
        for (let i = 0; i < BAR_COUNT; i++) {
            const bar = document.createElement('div');
            bar.classList.add('bar');
            visualizer.appendChild(bar);
            bars.push(bar);
        }

        // 2. 3D Tilt Interaction
        function handleTilt(x, y) {
            const rect = card.getBoundingClientRect();
            const centerX = rect.left + rect.width / 2;
            const centerY = rect.top + rect.height / 2;

            const rotateX = ((y - centerY) / (rect.height / 2)) * -18;
            const rotateY = ((x - centerX) / (rect.width / 2)) * 18;

            card.style.transform = `rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
        }

        window.addEventListener('mousemove', (e) => handleTilt(e.clientX, e.clientY));
        window.addEventListener('touchmove', (e) => {
            if (e.touches.length > 0) handleTilt(e.touches[0].clientX, e.touches[0].clientY);
        });
        window.addEventListener('mouseleave', () => {
            card.style.transform = `rotateX(0deg) rotateY(0deg)`;
        });

        // 3. Audio & Recording Engine
        let audioCtx;
        let analyser;
        let source;
        let micStream;
        let isRecordingMic = false;
        let isAudioPlaying = false;
        let wavePhase = 0;

        async function initAudioContext() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                analyser = audioCtx.createAnalyser();
                analyser.fftSize = 64;
                analyser.smoothingTimeConstant = 0.7; // Faster response for live voice
            }
            if (audioCtx.state === 'suspended') {
                await audioCtx.resume();
            }
        }

        // Loop for Voice & Music Wave
        function renderFrame() {
            if (!isRecordingMic && !isAudioPlaying) return;
            requestAnimationFrame(renderFrame);

            if (analyser) {
                const bufferLength = analyser.frequencyBinCount;
                const dataArray = new Uint8Array(bufferLength);
                analyser.getByteFrequencyData(dataArray);

                wavePhase += 0.1;

                for (let i = 0; i < BAR_COUNT; i++) {
                    const realValue = dataArray[i] || 0;
                    
                    // High gain boost for Mic input so voice creates noticeable big waves
                    let boostedVal = realValue * 2.8; 

                    // Idle background wave animation so bars never stay completely dead/flat
                    let idleWave = Math.sin(wavePhase + i * 0.4) * 8 + 10;
                    
                    let finalHeight = Math.max(boostedVal, idleWave);
                    finalHeight = Math.min(finalHeight, 130); // Max container limit

                    bars[i].style.height = `${finalHeight}px`;
                    bars[i].style.transform = `translateZ(${10 + (finalHeight * 0.15)}px)`;
                }
            }
        }

        function stopAllSources() {
            isRecordingMic = false;
            isAudioPlaying = false;

            if (audioElement && !audioElement.paused) {
                audioElement.pause();
            }

            if (micStream) {
                micStream.getTracks().forEach(track => track.stop());
                micStream = null;
            }

            if (source) {
                try { source.disconnect(); } catch (e) {}
            }

            playBtn.textContent = '▶';
            micBtn.classList.remove('btn-active');
            trackName.classList.remove('recording-active');

            // Reset visualizer bars
            bars.forEach(bar => {
                bar.style.height = '8px';
                bar.style.transform = 'translateZ(10px)';
            });
        }

        // 4. Local Audio File Upload
        audioUpload.addEventListener('change', async function(e) {
            const file = e.target.files[0];
            if (!file) return;

            await initAudioContext();
            stopAllSources();

            const cleanName = file.name.replace(/\.[^/.]+$/, "");
            trackName.textContent = `🎶 ${cleanName}`;

            const url = URL.createObjectURL(file);
            audioElement.src = url;

            source = audioCtx.createMediaElementSource(audioElement);
            source.connect(analyser);
            analyser.connect(audioCtx.destination);

            audioElement.play();
            playBtn.textContent = '⏸';
            isAudioPlaying = true;
            renderFrame();
        });

        // 5. Play / Pause Control
        playBtn.addEventListener('click', async () => {
            if (!audioElement.src) return;
            await initAudioContext();

            if (audioElement.paused) {
                stopAllSources();
                source = audioCtx.createMediaElementSource(audioElement);
                source.connect(analyser);
                analyser.connect(audioCtx.destination);

                audioElement.play();
                playBtn.textContent = '⏸';
                isAudioPlaying = true;
                renderFrame();
            } else {
                audioElement.pause();
                playBtn.textContent = '▶';
                isAudioPlaying = false;
            }
        });

        // 6. Voice Recording Handler
        micBtn.addEventListener('click', async () => {
            if (isRecordingMic) {
                stopAllSources();
                trackName.textContent = '🎵 Select Music or Use Mic';
            } else {
                try {
                    await initAudioContext();
                    stopAllSources();

                    micStream = await navigator.mediaDevices.getUserMedia({ 
                        audio: {
                            echoCancellation: true,
                            noiseSuppression: true,
                            autoGainControl: true
                        }, 
                        video: false 
                    });

                    source = audioCtx.createMediaStreamSource(micStream);
                    source.connect(analyser);

                    isRecordingMic = true;
                    micBtn.classList.add('btn-active');
                    trackName.textContent = '🔴 Recording Live Voice...';
                    trackName.classList.add('recording-active');

                    renderFrame();

                } catch (err) {
                    console.error("Mic Error:", err);
                    alert("Microphone permission allow karein!");
                }
            }
        });
    </script>
