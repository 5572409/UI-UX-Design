  <script>
    // ── CONFIG ──
    const CORRECT_OTP = '123456';
    let countdownTimer = null;
    let storedPhone = '';
    let isBreaking = false;
    let isPhoneVisible = false;

    // ── TOGGLE PHONE VISIBILITY ──
    function togglePhoneVisibility() {
      const input = document.getElementById('phoneInput');
      const eyeBtn = document.getElementById('eyeBtn');
      
      isPhoneVisible = !isPhoneVisible;
      
      if (isPhoneVisible) {
        input.type = 'text';
        eyeBtn.textContent = '👁‍🗨';
        eyeBtn.style.color = '#00ff41';
      } else {
        input.type = 'password';
        eyeBtn.textContent = '👁';
        eyeBtn.style.color = 'rgba(0, 255, 65, 0.3)';
      }
    }

    // ── INLINE TOAST ──
    function showToast(message, type = 'info', step = 1) {
      const toast = document.getElementById('toast' + step);
      toast.textContent = message;
      toast.className = `inline-toast ${type}`;
      toast.classList.add('show');
      
      clearTimeout(toast._hideTimer);
      toast._hideTimer = setTimeout(() => {
        toast.classList.remove('show');
      }, 3000);
    }

    // ── BLAST EFFECT ──
    function blastEffect() {
      const overlay = document.getElementById('blastOverlay');
      overlay.classList.remove('active');
      void overlay.offsetWidth;
      overlay.classList.add('active');
      setTimeout(() => overlay.classList.remove('active'), 900);
    }

    // ── CARD BREAK EFFECT ──
    function cardBreakEffect() {
      if (isBreaking) return;
      isBreaking = true;
      
      const card = document.getElementById('card');
      
      card.classList.remove('break', 'break-shake', 'red-flash');
      void card.offsetWidth;
      card.classList.add('break-shake');
      blastEffect();
      
      setTimeout(() => {
        card.classList.remove('break-shake');
        isBreaking = false;
      }, 900);
    }

    // ── STEP NAV ──
    function goStep(n) {
      document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
      document.getElementById('step' + n).classList.add('active');

      const dots = [document.getElementById('dot1'), document.getElementById('dot2'), document.getElementById('dot3')];
      dots.forEach((d, i) => {
        d.className = 'step-dot';
        if (i + 1 < n) d.classList.add('done');
        if (i + 1 === n) d.classList.add('active');
      });
    }

    // ── SEND OTP ──
    function sendOTP() {
      const raw = document.getElementById('phoneInput').value.trim();

      if (!raw) {
        showToast('Enter phone number', 'error', 1);
        return;
      }
      if (raw.length < 10) {
        showToast('Enter complete number (10 digits)', 'error', 1);
        return;
      }
      if (!raw.startsWith('3')) {
        showToast('Number must start with 3', 'error', 1);
        return;
      }

      storedPhone = '+92 ' + raw;
      document.getElementById('otpSubtitle').innerHTML = `Code sent to <span>${storedPhone}</span>`;

      ['o1','o2','o3','o4','o5','o6'].forEach(id => {
        const el = document.getElementById(id);
        el.value = '';
        el.classList.remove('filled');
      });

      goStep(2);
      startCountdown(30);
      setTimeout(() => document.getElementById('o1').focus(), 100);
      showToast('OTP sent successfully! ✨', 'success', 2);
    }

    // ── OTP BOX LOGIC ──
    function otpInput(current, nextId) {
      current.value = current.value.replace(/[^0-9]/g, '');
      if (current.value.length === 1) {
        current.classList.add('filled');
        if (nextId) document.getElementById(nextId).focus();
      } else {
        current.classList.remove('filled');
      }
    }

    function otpBack(event, prevId) {
      if (event.key === 'Backspace' && event.target.value === '' && prevId) {
        document.getElementById(prevId).focus();
      }
    }

    // ── VERIFY OTP ──
    function verifyOTP() {
      const entered = ['o1','o2','o3','o4','o5','o6'].map(id => document.getElementById(id).value).join('');

      if (entered.length < 6) {
        showToast('Enter full 6-digit OTP', 'error', 2);
        shakeOTP();
        return;
      }

      if (entered === CORRECT_OTP) {
        clearInterval(countdownTimer);
        showToast('OTP verified! 🎉', 'success', 2);
        setTimeout(() => {
          document.getElementById('step3').querySelector('.success-screen').style.display = 'block';
          goStep(3);
        }, 600);
      } else {
        showToast('Invalid OTP! Try again', 'error', 2);
        cardBreakEffect();
        shakeOTP();
        ['o1','o2','o3','o4','o5','o6'].forEach(id => {
          document.getElementById(id).value = '';
          document.getElementById(id).classList.remove('filled');
        });
        setTimeout(() => {
          document.getElementById('o1').focus();
        }, 200);
      }
    }

    function shakeOTP() {
      const wrap = document.getElementById('otpWrap');
      wrap.classList.remove('shake');
      void wrap.offsetWidth;
      wrap.classList.add('shake');
      setTimeout(() => wrap.classList.remove('shake'), 600);
    }

    // ── COUNTDOWN ──
    function startCountdown(seconds) {
      clearInterval(countdownTimer);
      const display = document.getElementById('countdownDisplay');
      const resendBtn = document.getElementById('resendBtn');
      resendBtn.classList.remove('active');
      resendBtn.style.cursor = 'not-allowed';

      let remaining = seconds;
      display.textContent = remaining + 's';

      countdownTimer = setInterval(() => {
        remaining--;
        display.textContent = remaining + 's';
        if (remaining <= 0) {
          clearInterval(countdownTimer);
          display.textContent = '00';
          resendBtn.classList.add('active');
          resendBtn.style.cursor = 'pointer';
        }
      }, 1000);
    }

    // ── RESEND ──
    function resendOTP() {
      const btn = document.getElementById('resendBtn');
      if (!btn.classList.contains('active')) return;

      ['o1','o2','o3','o4','o5','o6'].forEach(id => {
        document.getElementById(id).value = '';
        document.getElementById(id).classList.remove('filled');
      });

      startCountdown(30);
      document.getElementById('o1').focus();
      showToast('OTP resent! 🔄', 'info', 2);
    }

    // ── GO BACK ──
    function goBack() {
      clearInterval(countdownTimer);
      goStep(1);
    }

    // ── RESET ──
    function resetAll() {
      clearInterval(countdownTimer);
      document.getElementById('phoneInput').value = '';
      document.getElementById('step3').querySelector('.success-screen').style.display = 'none';
      goStep(1);
    }

    // ── ENTER KEY SUPPORT ──
    document.getElementById('phoneInput').addEventListener('keypress', function(e) {
      if (e.key === 'Enter') sendOTP();
    });

    document.querySelectorAll('.otp-box').forEach((box) => {
      box.addEventListener('keypress', function(e) {
        if (e.key === 'Enter') verifyOTP();
      });
    });

    // ── PREVENT AUTOFILL COMPLETELY ──
    document.addEventListener('DOMContentLoaded', function() {
      const phoneInput = document.getElementById('phoneInput');
      
      // Additional autofill prevention
      phoneInput.setAttribute('autocomplete', 'off');
      phoneInput.setAttribute('readonly', 'readonly');
      
      // Remove readonly on focus so user can type
      phoneInput.addEventListener('focus', function() {
        this.removeAttribute('readonly');
      });
      
      // Re-add readonly if input loses focus and is empty
      phoneInput.addEventListener('blur', function() {
        if (this.value === '') {
          this.setAttribute('readonly', 'readonly');
        }
      });
    });
  </script>
