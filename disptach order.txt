const orderBtn = document.getElementById('orderBtn');
const btnText = document.querySelector('.btn-text');

orderBtn.addEventListener('click', () => {
    if (orderBtn.classList.contains('animating') || orderBtn.classList.contains('done')) {
        return;
    }

    // Reset layout visibility just in case
    btnText.style.display = 'block';

    // Start Animation
    orderBtn.classList.add('animating');

    // Switch states strictly after 4.2s when van exits
    setTimeout(() => {
        orderBtn.classList.remove('animating');
        btnText.style.display = 'none'; // Hard remove dispatch text from viewport
        orderBtn.classList.add('done');
    }, 4200);

    // Auto-Reset loop back to original state after 8 seconds
    setTimeout(() => {
        orderBtn.classList.remove('done');
        btnText.style.display = 'block';
    }, 8000);
});
