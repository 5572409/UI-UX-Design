    <script>
        // DOM Elements
        const previewBox = document.getElementById('previewBox');
        const statusTitle = document.getElementById('statusTitle');
        const statusDesc = document.getElementById('statusDesc');
        const widthSlider = document.getElementById('widthSlider');
        const radiusSlider = document.getElementById('radiusSlider'); 
        const colorPicker = document.getElementById('colorPicker');
        
        const widthVal = document.getElementById('widthVal');
        const radiusVal = document.getElementById('radiusVal');
        
        const styleButtons = document.querySelectorAll('.style-btn');

        // State Variables
        let currentStyle = 'solid';
        let currentWidth = 9;
        let currentRadius = 8;
        let currentColor = '#00d2ff';

        // Update Function
        function updateStyles() {
            previewBox.style.borderStyle = currentStyle;
            previewBox.style.borderWidth = `${currentWidth}px`;
            previewBox.style.borderRadius = `${currentRadius}px`;
            previewBox.style.borderColor = currentColor;
            
            // Dynamic subtle light glow based on selected color
            previewBox.style.boxShadow = `0 10px 20px ${currentColor}25`; 

            widthVal.textContent = `${currentWidth}px`;
            radiusVal.textContent = `${currentRadius}px`;
        }

        // Button Click Event Listeners
        styleButtons.forEach(button => {
            button.addEventListener('click', () => {
                document.querySelector('.style-btn.active').classList.remove('active');
                button.classList.add('active');
                
                currentStyle = button.getAttribute('data-style');
                statusTitle.textContent = currentStyle;
                statusDesc.textContent = button.getAttribute('data-desc');
                
                updateStyles();
            });
        });

        // Sliders and Color Picker Listeners
        widthSlider.addEventListener('input', (e) => {
            currentWidth = e.target.value;
            updateStyles();
        });

        radiusSlider.addEventListener('input', (e) => {
            currentRadius = e.target.value;
            updateStyles();
        });

        colorPicker.addEventListener('input', (e) => {
            currentColor = e.target.value;
            updateStyles();
        });

        // Initialize App
        updateStyles();
    </script>
