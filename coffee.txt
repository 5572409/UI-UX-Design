          <span class="button-content">
            <svg class="cup-button-icon" viewBox="0 0 24 24" aria-hidden="true"><path d="M6 5h10l-1 15H7L6 5Z"/><path d="M5 5h12M8 9h6"/></svg>
            <span class="loader" aria-hidden="true"></span>
            <svg class="cart-button-icon" viewBox="0 0 24 24" aria-hidden="true" style="display:none"><path d="M4 5h2l2 10h9l2-7H7"/><circle cx="10" cy="19" r="1"/><circle cx="17" cy="19" r="1"/></svg>
            <span id="buttonText">Tap to fill</span>
          </span>
        </button>
      </div>
    </section>
  </main>
  <div class="toast" id="toast" role="status" aria-live="polite">Added to your bag</div>

  <script>
    (() => {
      const body = document.body;
      const machine = document.getElementById('machine');
      const cupWrap = document.getElementById('cupWrap');
      const machineStatus = document.getElementById('machineStatus');
      const sizeName = document.getElementById('sizeName');
      const priceEl = document.getElementById('price');
      const unitPriceEl = document.getElementById('unitPrice');
      const quantityEl = document.getElementById('quantity');
      const fillButton = document.getElementById('fillButton');
      const buttonText = document.getElementById('buttonText');
      const cupButtonIcon = document.querySelector('.cup-button-icon');
      const cartButtonIcon = document.querySelector('.cart-button-icon');
      const themeToggle = document.getElementById('themeToggle');
      const themeIcon = document.getElementById('themeIcon');
      const toast = document.getElementById('toast');
      const sizeButtons = [...document.querySelectorAll('.size-option')];

      let unitPrice = 30;
      let quantity = 1;
      let state = 'empty';
      let fillTimer;
      let toastTimer;

      function money(value) { return `$${value.toFixed(2)}`; }

      function updatePrice() {
        priceEl.textContent = money(unitPrice * quantity);
        unitPriceEl.innerHTML = quantity > 1 ? `${money(unitPrice)} × ${quantity}` : '&nbsp;';
      }

      function resetFill() {
        clearTimeout(fillTimer);
        state = 'empty';
        machine.classList.remove('pouring');
        cupWrap.classList.remove('filled');
        fillButton.classList.remove('filling', 'ready');
        fillButton.disabled = false;
        buttonText.textContent = 'Tap to fill';
        cupButtonIcon.style.display = '';
        cartButtonIcon.style.display = 'none';
        machineStatus.textContent = 'Ready';
      }

      sizeButtons.forEach((button) => {
        button.addEventListener('click', () => {
          sizeButtons.forEach((item) => {
            const active = item === button;
            item.classList.toggle('active', active);
            item.setAttribute('aria-pressed', String(active));
          });
          unitPrice = Number(button.dataset.price);
          sizeName.textContent = button.dataset.name;
          cupWrap.style.setProperty('--cup-w', `${button.dataset.width}px`);
          cupWrap.style.setProperty('--cup-h', `${button.dataset.height}px`);
          resetFill();
          updatePrice();
        });
      });

      document.getElementById('decrease').addEventListener('click', () => {
        quantity = Math.max(1, quantity - 1);
        quantityEl.textContent = quantity;
        updatePrice();
      });

      document.getElementById('increase').addEventListener('click', () => {
        quantity = Math.min(9, quantity + 1);
        quantityEl.textContent = quantity;
        updatePrice();
      });

      fillButton.addEventListener('click', () => {
        if (state === 'ready') {
          toast.classList.add('show');
          clearTimeout(toastTimer);
          toastTimer = setTimeout(() => toast.classList.remove('show'), 1800);
          return;
        }
        if (state !== 'empty') return;

        state = 'filling';
        fillButton.disabled = true;
        fillButton.classList.add('filling');
        machine.classList.remove('pouring');
        cupWrap.classList.remove('filled');
        void machine.offsetWidth;
        machine.classList.add('pouring');
        cupWrap.classList.add('filled');
        buttonText.textContent = 'Pouring…';
        machineStatus.textContent = 'Brewing';

        fillTimer = setTimeout(() => {
          state = 'ready';
          machine.classList.remove('pouring');
          fillButton.classList.remove('filling');
          fillButton.classList.add('ready');
          fillButton.disabled = false;
          buttonText.textContent = 'Add to cart';
          cupButtonIcon.style.display = 'none';
          cartButtonIcon.style.display = '';
          machineStatus.textContent = 'Complete';
        }, 2600);
      });

      themeToggle.addEventListener('click', () => {
        const isDark = body.classList.toggle('dark');
        themeToggle.setAttribute('aria-label', isDark ? 'Switch to light mode' : 'Switch to dark mode');
        themeIcon.innerHTML = isDark
          ? '<circle cx="12" cy="12" r="4"/><path d="M12 2v2M12 20v2M4.93 4.93l1.42 1.42M17.66 17.66l1.41 1.41M2 12h2M20 12h2M4.93 19.07l1.42-1.42M17.66 6.34l1.41-1.41"/>'
          : '<path d="M20.4 14.6A8.5 8.5 0 0 1 9.4 3.6 8.5 8.5 0 1 0 20.4 14.6Z"/>';
      });

      document.getElementById('cartButton').addEventListener('click', () => {
        toast.textContent = state === 'ready' ? 'Your drink is ready to add' : 'Fill your cup first';
        toast.classList.add('show');
        clearTimeout(toastTimer);
        toastTimer = setTimeout(() => {
          toast.classList.remove('show');
          setTimeout(() => { toast.textContent = 'Added to your bag'; }, 300);
        }, 1600);
      });

      updatePrice();
    })();
  </script>
</body>
