    shapes.forEach(s=>{s.mesh.rotation.x+=s.speed;s.mesh.rotation.y+=s.speed*.6;s.mesh.position.y+=Math.sin(t+s.float)*0.02;});
    cam.position.x+=(mx*5-cam.position.x)*.02;
    cam.position.y+=(-my*4-cam.position.y)*.02;
    cam.lookAt(scene.position);
    renderer.render(scene,cam);
  })();
  window.addEventListener('resize',()=>{
    cam.aspect=innerWidth/innerHeight;cam.updateProjectionMatrix();
    renderer.setSize(innerWidth,innerHeight);
  });
})();

/* ═══ GLOBE ═══ */
(()=>{
  const gc=document.getElementById('globe-canvas');
  const gctx=gc.getContext('2d');
  const W=400,H=400,R=160;
  let gt=0;
  const dots=[];
  for(let i=0;i<600;i++){
    const phi=Math.acos(-1+2*i/600);
    const theta=Math.sqrt(600*Math.PI)*phi;
    dots.push({phi,theta});
  }
  function drawGlobe(){
    gctx.clearRect(0,0,W,H);
    const cx=W/2,cy=H/2;
    /* outer glow */
    const gr=gctx.createRadialGradient(cx,cy,R*.7,cx,cy,R*1.2);
    gr.addColorStop(0,'rgba(124,58,237,0.15)');gr.addColorStop(1,'rgba(0,255,198,0)');
    gctx.beginPath();gctx.arc(cx,cy,R*1.2,0,Math.PI*2);
    gctx.fillStyle=gr;gctx.fill();
    /* equator */
    gctx.beginPath();gctx.arc(cx,cy,R,0,Math.PI*2);
    gctx.strokeStyle='rgba(124,58,237,0.15)';gctx.lineWidth=1;gctx.stroke();
    /* dots */
    dots.forEach(d=>{
      const phi=d.phi,theta=d.theta+gt;
      const x=R*Math.sin(phi)*Math.cos(theta);
      const y=R*Math.cos(phi);
      const z=R*Math.sin(phi)*Math.sin(theta);
      if(z>0){
        const sz=1.5+(z/R);
        const alpha=0.3+(z/R)*0.7;
        const grad=gctx.createRadialGradient(cx+x,cy+y,0,cx+x,cy+y,sz*2);
        grad.addColorStop(0,`rgba(0,255,198,${alpha})`);
        grad.addColorStop(1,'rgba(0,255,198,0)');
        gctx.beginPath();gctx.arc(cx+x,cy+y,sz,0,Math.PI*2);
        gctx.fillStyle=grad;gctx.fill();
      }
    });
    /* meridians */
    [0,Math.PI/4,Math.PI/2,3*Math.PI/4].forEach(off=>{
      gctx.beginPath();
      for(let a=0;a<=Math.PI*2;a+=0.05){
        const x=R*Math.cos(a+gt+off);const y=R*Math.sin(a);
        if(a===0)gctx.moveTo(cx+x,cy+y);else gctx.lineTo(cx+x,cy+y);
      }
      gctx.strokeStyle='rgba(124,58,237,0.1)';gctx.lineWidth=0.5;gctx.stroke();
    });
    gt+=0.005;
    requestAnimationFrame(drawGlobe);
  }
  drawGlobe();
})();

/* ═══ COUNTER ═══ */
const co=new IntersectionObserver(en=>{
  en.forEach(e=>{
    if(e.isIntersecting){
      const el=e.target,t=parseInt(el.dataset.c),step=t/60;
      let v=0;
      const iv=setInterval(()=>{v+=step;if(v>=t){v=t;clearInterval(iv);}el.textContent=Math.round(v)+'+';},25);
      co.unobserve(el);
    }
  });
},{threshold:0.5});
document.querySelectorAll('[data-c]').forEach(el=>co.observe(el));
