script::


const pages = [
    document.getElementById("p1"),
    document.getElementById("p2"),
    document.getElementById("p3")
];

const book = document.getElementById("book");
const maxLocation = pages.length + 1;
let currentLocation = 1;

pages.forEach((page, index) => {
    page.addEventListener("click", () => {
        const pageNum = index + 1;

        if (pageNum === currentLocation) {
            // Forward Flip
            page.classList.add("flipped");
            currentLocation++;
            
            setTimeout(() => {
                page.style.zIndex = pageNum;
            }, 300);

            // Positioning Logic
            if(pageNum === 1) {
                // Pehla page khulne par standard split view alignment
                const isMobile = window.innerWidth < 768;
                book.style.transform = isMobile ? "translateX(20%)" : "translateX(40%)";
            } else if (pageNum === pages.length) {
                // FIXED: Jab last page flip ho jaye, toh right side par positive shift denge 
                // taaki left side pe khula hua "Thank You" page wapis screen ke center mein aa jaye
                const isMobile = window.innerWidth < 768;
                book.style.transform = isMobile ? "translateX(50%)" : "translateX(50%)";
            }

        } else if (pageNum === currentLocation - 1) {
            // Backward Flip
            page.classList.remove("flipped");
            currentLocation--;

            setTimeout(() => {
                page.style.zIndex = maxLocation - pageNum;
            }, 100);

            if(pageNum === 1) {
                // Cover page standard center position
                book.style.transform = "translateX(0%)";
            } else if (pageNum === pages.length - 1) {
                // Agar aakhri page se back aa rahe hain, toh wapis normal split open alignment par le ayen
                const isMobile = window.innerWidth < 768;
                book.style.transform = isMobile ? "translateX(20%)" : "translateX(40%)";
            }
        }
    });
});
