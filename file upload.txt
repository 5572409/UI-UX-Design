  <script>
    const card = document.getElementById('card');
    const dropArea = document.getElementById('drop-area');
    const fileInput = document.getElementById('file-input');
    const fileList = document.getElementById('file-list');
    const footerStatus = document.getElementById('footer-status');
    const clearBtn = document.getElementById('clear-btn');

    let filesArray = [];
    let lastDropCoords = null;

    // Trigger browsing
    dropArea.addEventListener('click', () => fileInput.click());

    // Drag states
    dropArea.addEventListener('dragover', (e) => {
      e.preventDefault();
      dropArea.classList.add('drag-over');
    });

    dropArea.addEventListener('dragleave', () => {
      dropArea.classList.remove('drag-over');
    });

    dropArea.addEventListener('drop', (e) => {
      e.preventDefault();
      dropArea.classList.remove('drag-over');
      
      lastDropCoords = { x: e.clientX, y: e.clientY };
      handleFiles(e.dataTransfer.files);
    });

    fileInput.addEventListener('change', (e) => {
      const rect = dropArea.getBoundingClientRect();
      lastDropCoords = { x: rect.left + rect.width / 2, y: rect.top + rect.height / 2 };
      handleFiles(e.target.files);
    });

    function handleFiles(files) {
      if (filesArray.length >= 4) return; // Limit to max 4 on half-size card

      Array.from(files).forEach((file) => {
        if (filesArray.length >= 4) return;

        const fileId = 'f-' + Math.random().toString(36).substr(2, 9);
        const fileObj = {
          id: fileId,
          name: file.name,
          size: formatBytes(file.size),
          status: 'uploading',
          imgSrc: 'https://images.unsplash.com/photo-1542291026-7eec264c27ff?w=80&auto=format&fit=crop&q=80' 
        };

        if (file.type.startsWith('image/')) {
          const reader = new FileReader();
          reader.onload = (e) => {
            const ghost = document.getElementById(`ghost-${fileId}`);
            const live = document.getElementById(`live-${fileId}`);
            if (ghost) ghost.src = e.target.result;
            if (live) live.src = e.target.result;
          };
          reader.readAsDataURL(file);
        }

        filesArray.push(fileObj);
        renderFile(fileObj);
      });
      updateStatus();
    }

    function renderFile(fileObj) {
      const el = document.createElement('div');
      el.classList.add('file-item');
      el.id = `item-${fileObj.id}`;
      
      el.innerHTML = `
        <div class="preview-frame">
          <img class="img-ghost" id="ghost-${fileObj.id}" src="${fileObj.imgSrc}" alt="">
          <img class="img-live" id="live-${fileObj.id}" src="${fileObj.imgSrc}" alt="">
          <div class="img-percentage" id="percent-${fileObj.id}">0%</div>
        </div>
        <div class="file-info">
          <div class="file-name">${fileObj.name}</div>
          <div class="file-meta">
            <span id="badge-${fileObj.id}">0%</span>
          </div>
          <div class="progress-track">
            <div class="progress-fill" id="fill-${fileObj.id}"></div>
          </div>
        </div>
        <div id="action-slot-${fileObj.id}">
          <button class="action-btn" onclick="deleteItem('${fileObj.id}')">
            <svg xmlns="http://www.w3.org/2000/svg" width="8" height="8" fill="none" viewBox="0 0 24 24" stroke="currentColor" stroke-width="3">
              <path stroke-linecap="round" stroke-linejoin="round" d="M6 18L18 6M6 6l12 12" />
            </svg>
          </button>
        </div>
      `;

      fileList.appendChild(el);

      // --- SPRING ENTRANCE TRAJECTORY ---
      const itemRect = el.getBoundingClientRect();
      if (lastDropCoords) {
        const startX = lastDropCoords.x - itemRect.left - 15;
        const startY = lastDropCoords.y - itemRect.top - 15;
        
        el.animate([
          { transform: `translate(${startX}px, ${startY}px) scale(0.3)`, opacity: 0 },
          { transform: 'translate(0, 0) scale(1)', opacity: 1 }
        ], {
          duration: 550,
          easing: 'cubic-bezier(0.175, 0.885, 0.32, 1.2)'
        });
      }

      startLinearLiquidUpload(fileObj);
    }

    // Sequence progress upload with linear timer
    function startLinearLiquidUpload(fileObj) {
      const fill = document.getElementById(`fill-${fileObj.id}`);
      const badge = document.getElementById(`badge-${fileObj.id}`);
      const percentLabel = document.getElementById(`percent-${fileObj.id}`);
      const liveImg = document.getElementById(`live-${fileObj.id}`);
      const actionSlot = document.getElementById(`action-slot-${fileObj.id}`);

      let currentPercent = 0;
      const totalSteps = 100;
      const intervalTime = 25; // Super quick & smooth

      const timer = setInterval(() => {
        currentPercent += 1; // Pure 1% sequence increments

        if (currentPercent >= totalSteps) {
          currentPercent = totalSteps;
          clearInterval(timer);

          // Complete State
          badge.textContent = "Done";
          badge.className = "status-complete";
          percentLabel.classList.add('hidden');
          
          actionSlot.innerHTML = `
            <div class="check-success">
              <svg xmlns="http://www.w3.org/2000/svg" width="8" height="8" fill="none" viewBox="0 0 24 24" stroke="currentColor" stroke-width="3">
                <path stroke-linecap="round" stroke-linejoin="round" d="M5 13l4 4L19 7" />
              </svg>
            </div>
          `;
          
          fileObj.status = 'uploaded';
          updateStatus();
          
          // PREMIUM TRIGGER: Confetti Explosion!
          triggerConfetti(actionSlot);
        } else {
          percentLabel.textContent = `${currentPercent}%`;
          badge.textContent = `${currentPercent}%`;
        }

        fill.style.width = `${currentPercent}%`;

        // Wave Shift positioning
        const maskPosition = 100 - currentPercent;
        liveImg.style.webkitMaskPosition = `0% ${maskPosition}%`;
        liveImg.style.maskPosition = `0% ${maskPosition}%`;

      }, intervalTime);
    }

    // PREMIUM SUCCESS ANIMATION: Micro Confetti Burst
    function triggerConfetti(target) {
      const targetRect = target.getBoundingClientRect();
      const cardRect = card.getBoundingClientRect();
      
      const originX = targetRect.left - cardRect.left + (targetRect.width / 2);
      const originY = targetRect.top - cardRect.top + (targetRect.height / 2);

      const colors = ['#8a5cf5', '#10b981', '#ff5e57', '#ffca28', '#29b6f6'];

      for (let i = 0; i < 24; i++) {
        const particle = document.createElement('div');
        particle.classList.add('confetti');
        particle.style.background = colors[Math.floor(Math.random() * colors.length)];
        particle.style.left = `${originX}px`;
        particle.style.top = `${originY}px`;
        
        card.appendChild(particle);

        const angle = Math.random() * Math.PI * 2;
        const velocity = 25 + Math.random() * 45;
        const targetX = Math.cos(angle) * velocity;
        const targetY = Math.sin(angle) * velocity;

        particle.animate([
          { transform: 'translate(0, 0) scale(1)', opacity: 1 },
          { transform: `translate(${targetX}px, ${targetY}px) scale(0.3)`, opacity: 0 }
        ], {
          duration: 600 + Math.random() * 400,
          easing: 'cubic-bezier(0.1, 0.8, 0.25, 1)'
        }).onfinish = () => {
          particle.remove();
        };
      }
    }

    window.deleteItem = function(id) {
      filesArray = filesArray.filter(f => f.id !== id);
      const el = document.getElementById(`item-${id}`);
      if (el) {
        el.animate([
          { opacity: 1, transform: 'scale(1)' },
          { opacity: 0, transform: 'scale(0.8) translateY(-10px)' }
        ], { duration: 180, easing: 'ease' }).onfinish = () => {
          el.remove();
          updateStatus();
        };
      }
    };

    clearBtn.addEventListener('click', () => {
      filesArray = [];
      fileList.innerHTML = '';
      updateStatus();
    });

    function updateStatus() {
      const total = filesArray.length;
      const uploaded = filesArray.filter(f => f.status === 'uploaded').length;
      footerStatus.textContent = `${uploaded} of ${total} uploaded`;
    }

    function formatBytes(bytes) {
      if (bytes === 0) return '0 B';
      const k = 1024;
      const sizes = ['B', 'KB', 'MB'];
      const i = Math.floor(Math.log(bytes) / Math.log(k));
      return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i];
    }
  </script>
