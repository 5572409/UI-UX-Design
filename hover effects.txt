    <!-- ==========================================
       JAVASCRIPT FOR HOVER LAUNCH & 1-SEC STATE LOCK
       ========================================== -->
    <script>
        const buttons = document.querySelectorAll('.glass-btn:not(#tiltCard)');
        const container = document.getElementById('mainContainer');

        // Mouse Enter (Hover) Event for 1-Second Persistence
        buttons.forEach(btn => {
            btn.addEventListener('mouseenter', () => {
                // Effect trigger aur class add karna
                btn.classList.add('active');
                
                if(btn.classList.contains('effect-blur')) {
                    container.classList.add('blur-mode');
                }

                // 1 Second tak effect video recording ke liye locked rahega
                setTimeout(() => {
                    btn.classList.remove('active');
                    if(btn.classList.contains('effect-blur')) {
                        container.classList.remove('blur-mode');
                    }
                }, 1000);
            });
        });

        // 8th Button: 3D Tilt Hover Lock Mechanism
        const tiltCard = document.getElementById('tiltCard');
        tiltCard.addEventListener('mouseenter', () => {
            tiltCard.classList.add('active');
            
            // 3D Matrix tilt animation 1 second lock ke sath
            tiltCard.style.transform = `perspective(300px) rotateX(-12deg) rotateY(12deg) scale(0.96)`;
            tiltCard.style.background = "rgba(255, 255, 255, 0.75)";

            setTimeout(() => {
                tiltCard.classList.remove('active');
                tiltCard.style.transform = `perspective(300px) rotateX(0deg) rotateY(0deg) scale(1)`;
                tiltCard.style.background = "rgba(255, 255, 255, 0.45)";
            }, 1000);
        });
    </script>
