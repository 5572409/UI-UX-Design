      const body=document.body,product=document.getElementById('productVisual'),priceEl=document.getElementById('price'),selectionName=document.getElementById('selectionName'),fabricHint=document.getElementById('fabricHint'),fabricTag=document.getElementById('fabricTag'),colorLabel=document.getElementById('colorLabel'),button=document.getElementById('orderButton'),buttonText=document.getElementById('buttonText'),statusWrap=document.getElementById('statusWrap'),statusText=document.getElementById('statusText'),cartButton=document.getElementById('cartButton'),cartBadge=document.getElementById('cartBadge'),toast=document.getElementById('toast'),cartModal=document.getElementById('cartModal'),cartContent=document.getElementById('cartContent');
      const fabricButtons=[...document.querySelectorAll('.fabric-option')],colorButtons=[...document.querySelectorAll('.color-option')];
      let fabric={key:'cotton',name:'Cotton',title:'Cotton Essential',price:29,hint:'Soft everyday weave'},color={key:'black',name:'Matte Black'},state='idle',timers=[],changeTimers=[],toastTimer,lastOrder=null,cartCount=0;
      const colorHex={black:'#17191b',cream:'#eee6d7',sage:'#91a38c'};
      function setStatus(text,type=''){statusText.textContent=text;statusWrap.className=`status ${type}`}
      function clearTimers(list){list.forEach(clearTimeout);list.length=0}
      function flash(text){toast.textContent=text;toast.classList.add('show');clearTimeout(toastTimer);toastTimer=setTimeout(()=>toast.classList.remove('show'),1900)}
      function bumpPrice(){priceEl.classList.remove('bump');void priceEl.offsetWidth;priceEl.classList.add('bump')}
      function refreshUI(){priceEl.textContent=`$${fabric.price}`;selectionName.textContent=fabric.title;fabricHint.textContent=fabric.hint;fabricTag.textContent=`${fabric.name} · ${color.name}`;colorLabel.textContent=color.name;buttonText.textContent=`Add to Custom Order · $${fabric.price}`;bumpPrice()}
      function shimmer(apply){clearTimers(changeTimers);product.classList.remove('changing');void product.offsetWidth;product.classList.add('changing');changeTimers.push(setTimeout(apply,245));changeTimers.push(setTimeout(()=>product.classList.remove('changing'),730))}
      function resetOrder(){clearTimers(timers);state='idle';product.classList.remove('ordering','departing');button.classList.remove('processing','ready');button.disabled=false;setStatus('READY TO CUSTOMIZE');refreshUI();[...fabricButtons,...colorButtons].forEach(el=>el.disabled=false)}
      function flyToCart(){
        const source=product.getBoundingClientRect(),target=cartButton.getBoundingClientRect(),clone=product.cloneNode(true);clone.removeAttribute('id');clone.className='fly-shirt';clone.style.left=`${source.left}px`;clone.style.top=`${source.top}px`;clone.style.width=`${source.width}px`;clone.style.height=`${source.height}px`;document.body.appendChild(clone);product.classList.add('departing');
        const dx=target.left+target.width/2-(source.left+source.width/2),dy=target.top+target.height/2-(source.top+source.height/2);requestAnimationFrame(()=>requestAnimationFrame(()=>{clone.style.transform=`translate(${dx}px,${dy}px) scale(.1) rotate(9deg)`;clone.style.opacity='0';clone.style.filter='blur(2px)'}));
        setTimeout(()=>{clone.remove();cartButton.classList.add('receive');cartBadge.textContent=String(++cartCount);cartBadge.classList.add('show');setTimeout(()=>cartButton.classList.remove('receive'),600)},860)
      }
      function addOrder(){
        if(state==='processing')return;if(state==='ready'){resetOrder();return}state='processing';clearTimers(timers);button.disabled=true;button.classList.remove('ready');void button.offsetWidth;button.classList.add('processing');product.classList.add('ordering');[...fabricButtons,...colorButtons].forEach(el=>el.disabled=true);setStatus('PROCESSING ORDER...','processing');buttonText.textContent='✂ Tailoring Fabric...';
        timers.push(setTimeout(()=>buttonText.textContent='📦 Quality Check...',1450));
        timers.push(setTimeout(()=>{buttonText.textContent='Preparing your package...';flyToCart()},2700));
        timers.push(setTimeout(()=>{state='ready';lastOrder={fabric:fabric.name,color:color.name,price:fabric.price,colorHex:colorHex[color.key]};renderCart();button.classList.remove('processing');button.classList.add('ready');button.disabled=false;buttonText.textContent='✓ Ready to Ship';setStatus('ORDER READY','ready');product.classList.remove('ordering','departing');[...fabricButtons,...colorButtons].forEach(el=>el.disabled=false);flash('Custom T-shirt is ready to ship')},3700))
      }
      fabricButtons.forEach(btn=>btn.addEventListener('click',()=>{if(state==='processing')return;fabricButtons.forEach(item=>{const active=item===btn;item.classList.toggle('active',active);item.setAttribute('aria-pressed',String(active))});shimmer(()=>{fabric={key:btn.dataset.fabric,name:btn.dataset.name,title:btn.dataset.title,price:Number(btn.dataset.price),hint:btn.dataset.hint};body.dataset.fabric=fabric.key;resetOrder()})}));
      colorButtons.forEach(btn=>btn.addEventListener('click',()=>{if(state==='processing')return;colorButtons.forEach(item=>{const active=item===btn;item.classList.toggle('active',active);item.setAttribute('aria-pressed',String(active))});shimmer(()=>{color={key:btn.dataset.color,name:btn.dataset.name};body.dataset.color=color.key;resetOrder()})}));
      button.addEventListener('click',addOrder);
      function renderCart(){if(!lastOrder){cartContent.innerHTML='<div class="cart-empty">Your custom cart is empty.</div>';return}cartContent.innerHTML=`<div class="cart-item"><div class="mini-shirt" style="--ordered-color:${lastOrder.colorHex}"><svg viewBox="0 0 24 24" aria-hidden="true"><path d="M8 4c2 2 6 2 8 0l4 2 3 5-4 2-1-2v9H6v-9l-1 2-4-2 3-5z"/></svg></div><div class="cart-copy"><b>${lastOrder.fabric} Custom Tee</b><span>${lastOrder.color} · Ready to ship</span></div><span class="cart-price">$${lastOrder.price}</span></div>`}
      function openCart(){renderCart();cartModal.classList.add('open')}
      function closeCart(){cartModal.classList.remove('open')}
      cartButton.addEventListener('click',openCart);document.getElementById('closeCart').addEventListener('click',closeCart);cartModal.addEventListener('click',e=>{if(e.target===cartModal)closeCart()});document.addEventListener('keydown',e=>{if(e.key==='Escape')closeCart()});
      document.getElementById('themeToggle').addEventListener('click',()=>{const dark=body.classList.toggle('dark');document.querySelector('meta[name="theme-color"]').content=dark?'#0c0c0d':'#eeeae4';document.getElementById('themeToggle').setAttribute('aria-label',dark?'Switch to light mode':'Switch to dark mode');document.getElementById('themeIcon').innerHTML=dark?'<circle cx="12" cy="12" r="4"/><path d="M12 2v2M12 20v2M4.93 4.93l1.42 1.42M17.66 17.66l1.41 1.41M2 12h2M20 12h2M4.93 19.07l1.42-1.42M17.66 6.34l1.41-1.41"/>':'<path d="M20.4 14.6A8.5 8.5 0 0 1 9.4 3.6 8.5 8.5 0 1 0 20.4 14.6Z"/>'});
    })();
  </script>
</body>
</html>
