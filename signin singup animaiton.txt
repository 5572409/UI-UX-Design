class AuthAnimationApp extends StatelessWidget {

  Widget build(BuildContext context) {
    return const MaterialApp(
      debugShowCheckedModeBanner: false,
      home: AuthAnimationPage(),
    );
  }

class AuthAnimationPage extends StatefulWidget {
  const AuthAnimationPage({super.key});

  @override
  State<AuthAnimationPage> createState() => _AuthAnimationPageState();
}

class _AuthAnimationPageState extends State<AuthAnimationPage>
    with TickerProviderStateMixin {
  late final AnimationController _controller;
  late final AnimationController _clickController;

  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 1200),
    );
    _clickController = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 800),
    );
  }

  void _openSignup() {
    if (_controller.value == 1 || _controller.isAnimating) return;
    _clickController.forward().then((_) {
      _clickController.reset();
    });
    _controller.forward();
  }

  void _openSignin() {
    if (_controller.value == 0 || _controller.isAnimating) return;
    _clickController.forward().then((_) {
      _clickController.reset();
    });
    _controller.reverse();
  }

  @override
  void dispose() {
    _controller.dispose();
    _clickController.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: MouseRegion(
        cursor: SystemMouseCursors.basic,
        child: Container(
          decoration: const BoxDecoration(
            gradient: LinearGradient(
              begin: Alignment.topCenter,
              end: Alignment.bottomCenter,
              colors: [
                Color(0xFF0A0E27),
                Color(0xFF0F1A3A),
                Color(0xFF06081A),
              ],
            ),
          ),
          child: SafeArea(
            child: LayoutBuilder(
              builder: (context, constraints) {
                final isWebSize = constraints.maxWidth >= 720;
                final maxCardWidth = isWebSize ? 760.0 : 390.0;
                final sidePadding = constraints.maxWidth < 420 ? 16.0 : 28.0;
                final cardWidth = math.min(
                  constraints.maxWidth - (sidePadding * 2),
                  maxCardWidth,
                );

                return Center(
                  child: SingleChildScrollView(
                    padding: EdgeInsets.symmetric(
                      horizontal: sidePadding,
                      vertical: 24,
                    ),
                    child: Stack(
                      children: [
                        AnimatedBuilder(
                          animation: _clickController,
                          builder: (context, _) {
                            return Container(
                              width: cardWidth,
                              height: cardWidth / 1.9,
                              decoration: BoxDecoration(
                                borderRadius: BorderRadius.circular(8),
                                boxShadow: [
                                  BoxShadow(
                                    color: Color.lerp(
                                      const Color(0xFF00D9FF).withOpacity(0.0),
                                      const Color(0xFF00D9FF).withOpacity(0.6),
                                      _clickController.value,
                                    )!,
                                    blurRadius: 60 + (_clickController.value * 40),
                                    spreadRadius: 10 + (_clickController.value * 15),
                                  ),
                                  BoxShadow(
                                    color: Color.lerp(
                                      const Color(0xFF7C3AED).withOpacity(0.0),
                                      const Color(0xFF7C3AED).withOpacity(0.4),
                                      _clickController.value,
                                    )!,
                                    blurRadius: 50 + (_clickController.value * 30),
                                    spreadRadius: 5 + (_clickController.value * 12),
                                  ),
                                ],
                              ),
                            );
                          },
                        ),
                        AuthAnimationCard(
                          controller: _controller,
                          clickController: _clickController,
                          width: cardWidth,
                          onSignupTap: _openSignup,
                          onSigninTap: _openSignin,
                        ),
                      ],
                    ),
                  ),
                );
              },
            ),
          ),
        ),
      ),
    );
  }
}

class _AuthMotion {
  const _AuthMotion({
    required this.side,
    required this.loginOpacity,
    required this.registerOpacity,
    required this.sweep,
  });

  final double side;
  final double loginOpacity;
  final double registerOpacity;
  final double sweep;

  static _AuthMotion from(double t) {
    return _AuthMotion(
      side: _lerp(-1, 1, t),
      loginOpacity: 1 - _ease(_range(t, 0.00, 0.36)),
      registerOpacity: _ease(_range(t, 0.64, 1.00)),
      sweep: math.sin(t * math.pi),
    );
  }
}

class _AuthPanelPainter extends CustomPainter {
  const _AuthPanelPainter({
    required this.side,
    required this.sweep,
    required this.shift,
  });

  final double side;
  final double sweep;
  final double shift;

  @override
  void paint(Canvas canvas, Size size) {
    final rect = Offset.zero & size;

    // Enhanced gradient for 8K - richer colors
    final gradientPaint = Paint()
      ..shader = const LinearGradient(
        begin: Alignment.topLeft,
        end: Alignment.bottomRight,
        colors: [
          Color(0xFF1A2856),
          Color(0xFF2E3B7F),
          Color(0xFF3D51B5),
          Color(0xFF1E2847),
        ],
        stops: [0.0, 0.38, 0.68, 1.0],
      ).createShader(rect);
    canvas.drawRect(rect, gradientPaint);

    _drawTexture(canvas, size);
    _drawDarkShape(canvas, size);
    _drawSweep(canvas, size);
    _drawBorder(canvas, size);

    // Extra glow layer for 8K
    _drawGlow(canvas, size);
  }

  void _drawTexture(Canvas canvas, Size size) {
    final w = size.width;
    final h = size.height;
    final linePaint = Paint()
      ..color = Colors.white.withOpacity(0.06)
      ..strokeWidth = 0.8;
    final shadowPaint = Paint()
      ..color = Colors.black.withOpacity(0.15)
      ..strokeWidth = 1.5;

    final drift = shift * 34;
    for (double x = -h; x < w + h; x += 18) {
      final start = Offset(x + drift, 0);
      final end = Offset(x - h * 0.45 + drift, h);
      canvas.drawLine(start.translate(8, 0), end.translate(8, 0), shadowPaint);
      canvas.drawLine(start, end, linePaint);
    }

    final shade = Paint()
      ..shader = LinearGradient(
        begin: Alignment.topLeft,
        end: Alignment.bottomRight,
        colors: [
          Colors.white.withOpacity(0.08),
          Colors.transparent,
          Colors.black.withOpacity(0.2),
        ],
      ).createShader(Offset.zero & size);
    canvas.drawRect(Offset.zero & size, shade);
  }

  void _drawDarkShape(Canvas canvas, Size size) {
    final w = size.width;
    final h = size.height;
    final u = ((side + 1) / 2).clamp(0.0, 1.0).toDouble();

    final p0 = Offset(_lerp(0, w * 0.55, u), 0);
    final p1 = Offset(_lerp(w * 0.48, w, u), 0);
    final p2 = Offset(_lerp(w * 0.74, w, u), h);
    final p3 = Offset(_lerp(0, w * 0.28, u), h);

    final path = Path()
      ..moveTo(p0.dx, p0.dy)
      ..lineTo(p1.dx, p1.dy)
      ..lineTo(p2.dx, p2.dy)
      ..lineTo(p3.dx, p3.dy)
      ..close();

    canvas.drawShadow(path, Colors.black.withOpacity(0.8), 20, true);
    canvas.drawPath(
      path,
      Paint()
        ..shader = const LinearGradient(
          begin: Alignment.topLeft,
          end: Alignment.bottomRight,
          colors: [
            Color(0xFF0F1829),
            Color(0xFF1A2856),
            Color(0xFF0A0F1F),
          ],
        ).createShader(Offset.zero & size),
    );

    // Enhanced edge glow for 8K
    final edgeTop = Offset(_lerp(w * 0.48, w * 0.55, u), 0);
    final edgeBottom = Offset(_lerp(w * 0.74, w * 0.28, u), h);

    canvas.drawLine(
      edgeTop,
      edgeBottom,
      Paint()
        ..color = const Color(0xFF00D9FF).withOpacity(0.9)
        ..strokeWidth = 8
        ..maskFilter = const MaskFilter.blur(BlurStyle.normal, 12),
    );
    canvas.drawLine(
      edgeTop,
      edgeBottom,
      Paint()
        ..color = const Color(0xFF00E5FF)
        ..strokeWidth = 2,
    );
  }

  void _drawSweep(Canvas canvas, Size size) {
    if (sweep <= 0.01) return;

    final w = size.width;
    final h = size.height;
    final x = _lerp(-w * 0.35, w * 1.25, shift);

    final sweepPath = Path()
      ..moveTo(x - w * 0.28, 0)
      ..lineTo(x + w * 0.18, 0)
      ..lineTo(x - w * 0.28, h)
      ..lineTo(x - w * 0.74, h)
      ..close();

    canvas.drawPath(
      sweepPath,
      Paint()
        ..color = const Color(0xFF00D9FF).withOpacity(0.25 * sweep)
        ..maskFilter = const MaskFilter.blur(BlurStyle.normal, 24),
    );
    canvas.drawRect(
      Offset.zero & size,
      Paint()..color = const Color(0xFF00D9FF).withOpacity(0.08 * sweep),
    );
  }

  void _drawBorder(Canvas canvas, Size size) {
    final rect = Offset.zero & size;
    final rrect = RRect.fromRectAndRadius(
      rect.deflate(2),
      const Radius.circular(8),
    );

    canvas.drawRRect(
      rrect,
      Paint()
        ..style = PaintingStyle.stroke
        ..strokeWidth = 10
        ..color = const Color(0xFF00D9FF).withOpacity(0.2)
        ..maskFilter = const MaskFilter.blur(BlurStyle.normal, 15),
    );
    canvas.drawRRect(
      rrect,
      Paint()
        ..style = PaintingStyle.stroke
        ..strokeWidth = 2
        ..color = const Color(0xFF00D9FF).withOpacity(0.7),
    );
    canvas.drawRRect(
      rrect.deflate(4),
      Paint()
        ..style = PaintingStyle.stroke
        ..strokeWidth = 1
        ..color = Colors.white.withOpacity(0.2),
    );
  }

  void _drawGlow(Canvas canvas, Size size) {
    final rect = Offset.zero & size;
    final gradient = LinearGradient(
      begin: Alignment.topLeft,
      end: Alignment.bottomRight,
      colors: [
        const Color(0xFF00D9FF).withOpacity(0.03),
        Colors.transparent,
        const Color(0xFF7C3AED).withOpacity(0.03),
      ],
    );
    canvas.drawRect(rect, Paint()..shader = gradient.createShader(rect));
  }

  @override
  bool shouldRepaint(covariant _AuthPanelPainter oldDelegate) {
    return oldDelegate.side != side ||
        oldDelegate.sweep != sweep ||
        oldDelegate.shift != shift;
  }
}

class _FadeSlide extends StatelessWidget {
  const _FadeSlide({
    required this.opacity,
    required this.dx,
    required this.child,
  });

  final double opacity;
  final double dx;
  final Widget child;

  @override
  Widget build(BuildContext context) {
    final visible = opacity.clamp(0.0, 1.0).toDouble();

    return IgnorePointer(
      ignoring: visible < 0.2,
      child: Opacity(
        opacity: visible,
        child: Transform.translate(
          offset: Offset(dx * (1 - visible), 0),
          child: child,
        ),
      ),
    );
  }
}

class _GradientButton extends StatefulWidget {
  const _GradientButton({
    required this.text,
    required this.unit,
  });

  final String text;
  final double unit;

  @override
  State<_GradientButton> createState() => _GradientButtonState();
}

class _GradientButtonState extends State<_GradientButton>
    with SingleTickerProviderStateMixin {
  late AnimationController _pressController;
  bool _isPressed = false;

  @override
  void initState() {
    super.initState();
    _pressController = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 600),
    );
  }

  @override
  void dispose() {
    _pressController.dispose();
    super.dispose();
  }

  void _onPressed() {
    setState(() => _isPressed = true);
    _pressController.forward().then((_) {
      _pressController.reverse();
      setState(() => _isPressed = false);
    });
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: _pressController,
      builder: (context, _) {
        final pressValue = _pressController.value;

        return MouseRegion(
          cursor: SystemMouseCursors.click,
          child: GestureDetector(
            onTap: _onPressed,
            child: Container(
              height: 25 * widget.unit,
              alignment: Alignment.center,
              decoration: BoxDecoration(
                borderRadius: BorderRadius.circular(100),
                gradient: LinearGradient(
                  begin: Alignment.topCenter,
                  end: Alignment.bottomCenter,
                  colors: [
                    Color.lerp(
                      const Color(0xFF5EB3FF),
                      const Color(0xFF00D9FF),
                      pressValue,
                    )!,
                    Color.lerp(
                      const Color(0xFF3D51B5),
                      const Color(0xFF00D9FF).withOpacity(0.8),
                      pressValue,
                    )!,
                    Color.lerp(
                      const Color(0xFF2E3B7F),
                      const Color(0xFF7C3AED),
                      pressValue,
                    )!,
                  ],
                ),
                border: Border.all(
                  color: Colors.white.withOpacity(
                    0.25 + (pressValue * 0.35),

class _FooterLink extends StatefulWidget {
  const _FooterLink({
    required this.text,
    required this.unit,
    required this.onTap,
  });

  final String text;
  final double unit;
  final VoidCallback onTap;

  @override
  State<_FooterLink> createState() => _FooterLinkState();
}

class _FooterLinkState extends State<_FooterLink> {
  bool _isHovered = false;

  @override
  Widget build(BuildContext context) {
    return MouseRegion(
      cursor: SystemMouseCursors.click,
      onEnter: (_) => setState(() => _isHovered = true),
      onExit: (_) => setState(() => _isHovered = false),
      child: GestureDetector(
        behavior: HitTestBehavior.opaque,
        onTap: widget.onTap,
        child: Padding(
          padding: EdgeInsets.symmetric(vertical: 3 * widget.unit),
          child: AnimatedDefaultTextStyle(
            duration: const Duration(milliseconds: 300),
            style: TextStyle(
              color: Color.lerp(
                const Color(0xFF00D9FF),
                const Color(0xFF7C3AED),
                _isHovered ? 1.0 : 0.0,
              )!,
              fontSize: 8 * widget.unit,
              fontWeight: FontWeight.w900,
              letterSpacing: 0.5,
              shadows: [
                Shadow(
                  color: Colors.black.withOpacity(0.3),
                  blurRadius: 4,
                  offset: const Offset(0, 2),
                ),
              ],
            ),
            child: Text(
              widget.text,
              textAlign: TextAlign.center,
            ),
          ),
        ),
      ),
    );
  }
}

double _range(double value, double start, double end) {
  return ((value - start) / (end - start)).clamp(0.0, 1.0).toDouble();
}

double _ease(double value) {
  return Curves.easeInOutCubic.transform(value.clamp(0.0, 1.0).toDouble());
}

double _lerp(double a, double b, double t) {
  return a + (b - a) * t;
}
