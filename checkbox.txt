    <script>
        const splashInputs = document.querySelectorAll('.splash-input');

        splashInputs.forEach(input => {
            input.addEventListener('change', (event) => {
                if (event.target.checked) {
                    const ring = event.target.nextElementSibling.querySelector('.splash-ring');
                    if (ring) {
                        ring.classList.remove('ripple-active');
                        void ring.offsetWidth; 
                        ring.classList.add('ripple-active');
                    }
                }
            });
        });
