    <script>
        const display = document.getElementById('display');

        // Screen par character add krny k liye
        function appendValue(value) {
            if (display.value === 'Error') {
                display.value = '';
            }
            display.value += value;
        }

        // Pura display clear krny k liye
        function clearDisplay() {
            display.value = '';
        }

        // Single last character clear krny k liye (Backspace)
        function backspace() {
            display.value = display.value.slice(0, -1);
        }

        // Calculation logic
        function calculate() {
            try {
                if (display.value !== '') {
                    // eval mathematically input parse krta hai
                    display.value = eval(display.value);
                }
            } catch (error) {
                display.value = 'Error';
            }
        }
    </script>
