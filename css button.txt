    <script>
        // Dropdown Menu Toggle
        const dropdownBtn = document.getElementById('dropdownBtn');
        const dropdownMenu = document.getElementById('dropdownMenu');
        dropdownBtn.addEventListener('click', () => {
            dropdownMenu.classList.toggle('show');
        });

        // Dropdown close on outside click
        window.addEventListener('click', (e) => {
            if (!dropdownBtn.contains(e.target)) {
                dropdownMenu.classList.remove('show');
            }
        });

        // Toggle Button State
        const toggleBtn = document.getElementById('toggleBtn');
        toggleBtn.addEventListener('click', () => {
            toggleBtn.classList.toggle('btn-toggle-active');
            if (toggleBtn.classList.contains('btn-toggle-active')) {
                toggleBtn.innerText = "Active";
            } else {
                toggleBtn.innerText = "Toggle Mode";
            }
        });

        // Loading Spinner Simulation
        const loadingBtn = document.getElementById('loadingBtn');
        loadingBtn.addEventListener('click', () => {
            loadingBtn.classList.add('btn-loading');
            setTimeout(() => {
                loadingBtn.classList.remove('btn-loading');
                alert('Success! Data fetched.');
            }, 2000); // 2 second loading delay
        });
    </script>
