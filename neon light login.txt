const usernameInput = document.getElementById('username');
const passwordInput = document.getElementById('password');
const loginBtn = document.getElementById('loginButton');
const loginForm = document.getElementById('loginForm');
const statusMessage = document.getElementById('statusMessage');
const formTitle = document.getElementById('formTitle');

// Visual Overlays
const beams = document.querySelectorAll('.laser-beam');
const strobeOverlay = document.querySelector('.strobe-overlay');
const techRings = document.querySelectorAll('.ring');

// Checking continuous typing inputs to illuminate the master layout
function monitorSystemSync() {
    const userVal = usernameInput.value.trim();
    const passVal = passwordInput.value.trim();

    if (userVal.length > 0 || passVal.length > 0) {
        loginBtn.classList.add('btn-ready');
        loginBtn.innerText = "LAUNCH QUANTUM SYNC";
        
        // Speed up the ambient tech background rings
        techRings.forEach((ring, index) => {
            ring.style.animationDuration = `${(index + 1) * 3}s`;
        });
    } else {
        loginBtn.classList.remove('btn-ready');
        loginBtn.innerText = "ENGAGE QUANTUM LINK";
        
        // Return rings to normal cruise speeds
        techRings[0].style.animationDuration = "20s";
        techRings[1].style.animationDuration = "15s";
        techRings[2].style.animationDuration = "8s";
    }
}

usernameInput.addEventListener('input', monitorSystemSync);
passwordInput.addEventListener('input', monitorSystemSync);

// Master Blast Animation Engine & Auto Recovery
loginBtn.addEventListener('click', (e) => {
    e.preventDefault();
    
    const finalUser = usernameInput.value.trim();
    const finalPass = passwordInput.value.trim();

    if (finalUser !== "" && finalPass !== "") {
        // --- 1. THE HEAVY BLAST ANIMATION SEQUENCE ---
        formTitle.innerText = "LINK ESTABLISHED";
        formTitle.style.color = "#00ff66";
        formTitle.style.textShadow = "0 0 20px #00ff66";
        
        statusMessage.innerText = "SUCCESS: ACCESS GRANTED. SYNCHRONIZING DATAFEED...";
        statusMessage.style.color = "#00ff66";
        
        // Activate full screen hyper strobes and warp classes
        loginForm.classList.remove('hyper-glow');
        loginForm.classList.add('hyper-blast-active');

        beams.forEach(beam => beam.classList.add('beams-active'));
        strobeOverlay.classList.add('strobe-active');

        // --- 2. THE AUTOMATIC SYSTEM RESET TIME WINDOW ---
        setTimeout(() => {
            // Remove heavy hyper active visual FX
            beams.forEach(beam => beam.classList.remove('beams-active'));
            strobeOverlay.classList.remove('strobe-active');
            loginForm.classList.remove('hyper-blast-active');
            
            // Re-apply standard beautiful infinite rainbow fluid mode
            loginForm.classList.add('hyper-glow');
            
            formTitle.innerText = "CORE SYSTEM LINK";
            formTitle.style.color = "#fff";
            formTitle.style.textShadow = "0 0 10px rgba(255,255,255,0.4)";
            
            statusMessage.innerText = "";
            
            // Flush entries back to native fresh state
            usernameInput.value = "";
            passwordInput.value = "";
            monitorSystemSync();
            
        }, 3000); // 3 seconds window duration before restoring system equilibrium

    } else {
        // --- MATRIX SHAKE ERROR ENGINE ---
        loginForm.classList.add('shake-deny');
        statusMessage.innerText = "CRITICAL ERROR: ACCESS CREDENTIALS REQUIRED!";
        statusMessage.style.color = "#ff007f";
        
        setTimeout(() => {
            loginForm.classList.remove('shake-deny');
        }, 400);
    }
});
