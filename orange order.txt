  <script>
    (() => {
      const body = document.body;
      const scene = document.getElementById('scene');
      const cup = document.getElementById('cupWrap');
      const cupStamp = document.getElementById('cupStamp');
      const status = document.getElementById('machineStatus');
      const statusWrap = document.getElementById('statusWrap');
      const sizeName = document.getElementById('sizeName');
      const priceEl = document.getElementById('price');
      const unitEl = document.getElementById('unitPrice');
      const qtyEl = document.getElementById('quantity');
      const pressButton = document.getElementById('pressButton');
      const buttonText = document.getElementById('buttonText');
      const orangeIcon = document.querySelector('.orange-button-icon');
      const cartIcon = document.querySelector('.cart-button-icon');
      const toast = document.getElementById('toast');
      const cartBadge = document.getElementById('cartBadge');
      const cartModal = document.getElementById('cartModal');
      const cartContent = document.getElementById('cartContent');
      const checkoutDone = document.getElementById('checkoutDone');
      const sizeButtons = [...document.querySelectorAll('.size-option')];
      const choices = [...document.querySelectorAll('.choice')];

      let basePrice = 199;
      let volume = 250;
      let pulp = { name: 'Regular Pulp', price: 0 };
      let ice = { name: 'Regular Ice', price: 0 };
      let quantity = 1;
      let state = 'empty';
      let timers = [];
      let toastTimer;
      let cartCount = 0;
      let cartTotal = 0;
      let lastItem = null;

      const money = value => `Rs ${value.toLocaleString('en-PK')}`;
      const unitTotal = () => basePrice + pulp.price + ice.price;

      function updatePrice() {
        const unit = unitTotal();
        priceEl.textContent = money(unit * quantity);
        unitEl.textContent = quantity > 1 ? `${money(unit)} × ${quantity} · ${volume} ml each` : `${volume} ml`;
      }
      function clearTimers() { timers.forEach(clearTimeout); timers = []; }
      function setStatus(text, type = '') { status.textContent = text; statusWrap.className = `machine-status ${type}`; }
      function lockOptions(lock) {
        [...sizeButtons, ...choices, document.getElementById('decrease'), document.getElementById('increase')].forEach(element => { element.disabled = lock; });
      }
      function resetPress() {
        clearTimers();
        state = 'empty';
        scene.classList.remove('placing','loaded','pressing','complete');
        pressButton.classList.remove('working','ready');
        pressButton.disabled = false;
        buttonText.textContent = 'Press fresh juice';
        orangeIcon.style.display = '';
        cartIcon.style.display = 'none';
        setStatus('Ready');
      }
      function flash(message) {
        toast.textContent = message;
        toast.classList.add('show');
        clearTimeout(toastTimer);
        toastTimer = setTimeout(() => toast.classList.remove('show'), 1900);
      }

      sizeButtons.forEach(button => button.addEventListener('click', () => {
        sizeButtons.forEach(item => {
          const active = item === button;
          item.classList.toggle('active', active);
          item.setAttribute('aria-pressed', String(active));
        });
        basePrice = Number(button.dataset.price);
        volume = Number(button.dataset.volume);
        sizeName.textContent = button.dataset.name;
        cup.dataset.size = button.dataset.size;
        cupStamp.textContent = `${volume}ml`;
        scene.style.setProperty('--cup-w', `${button.dataset.width}px`);
        scene.style.setProperty('--cup-h', `${button.dataset.height}px`);
        scene.style.setProperty('--stream-h', `${button.dataset.stream}px`);
        resetPress();
        updatePrice();
      }));

      choices.forEach(choice => choice.addEventListener('click', () => {
        document.querySelectorAll(`.choice[data-type="${choice.dataset.type}"]`).forEach(item => item.classList.toggle('active', item === choice));
        const selected = { name: choice.dataset.name, price: Number(choice.dataset.price) };
        if (choice.dataset.type === 'pulp') pulp = selected;
        if (choice.dataset.type === 'ice') ice = selected;
        resetPress();
        updatePrice();
      }));

      document.getElementById('decrease').addEventListener('click', () => {
        quantity = Math.max(1, quantity - 1);
        qtyEl.textContent = quantity;
        updatePrice();
      });
      document.getElementById('increase').addEventListener('click', () => {
        quantity = Math.min(9, quantity + 1);
        qtyEl.textContent = quantity;
        updatePrice();
      });

      pressButton.addEventListener('click', () => {
        if (state === 'ready') {
          lastItem = { size: sizeName.textContent, volume, pulp: pulp.name, ice: ice.name, quantity, total: unitTotal() * quantity };
          cartCount += quantity;
          cartTotal += lastItem.total;
          cartBadge.textContent = cartCount;
          cartBadge.classList.add('show');
          renderCart();
          flash(`${quantity} fresh juice${quantity > 1 ? 's' : ''} added to your bag`);
          buttonText.textContent = 'Added ✓';
          timers.push(setTimeout(() => { buttonText.textContent = 'Add to cart'; }, 900));
          return;
        }
        if (state !== 'empty') return;

        state = 'working';
        lockOptions(true);
        pressButton.disabled = true;
        pressButton.classList.add('working');
        scene.classList.add('placing');
        setStatus('Placing orange','working');
        buttonText.textContent = 'Placing orange…';

        timers.push(setTimeout(() => {
          scene.classList.add('loaded');
          setStatus('Orange secured','working');
          buttonText.textContent = 'Orange inside press…';
        }, 1120));

        timers.push(setTimeout(() => {
          scene.classList.remove('placing');
          scene.classList.add('pressing');
          setStatus('Pressing by hand','working');
          buttonText.textContent = 'Manual pressing…';
        }, 1780));

        timers.push(setTimeout(() => {
          setStatus('Filling cup','working');
          buttonText.textContent = `Filling ${volume} ml cup…`;
        }, 2720));

        timers.push(setTimeout(() => {
          state = 'ready';
          scene.classList.add('complete');
          pressButton.classList.remove('working');
          pressButton.classList.add('ready');
          pressButton.disabled = false;
          lockOptions(false);
          buttonText.textContent = 'Add to cart';
          orangeIcon.style.display = 'none';
          cartIcon.style.display = '';
          setStatus('Fresh & ready','done');
        }, 5350));
      });

      function renderCart() {
        if (!lastItem) {
          cartContent.innerHTML = '<div class="cart-empty">Your bag is empty. Press a fresh juice first.</div>';
          return;
        }
        cartContent.innerHTML = `<div class="cart-item"><div class="cart-item-top"><b>${lastItem.size} × ${lastItem.quantity}</b><b>${money(lastItem.total)}</b></div><p>${lastItem.volume} ml · ${lastItem.pulp}<br>${lastItem.ice}</p></div><div class="cart-total"><span>Bag total (${cartCount} item${cartCount > 1 ? 's' : ''})</span><b>${money(cartTotal)}</b></div><button class="checkout-btn" id="checkoutButton">Place order</button>`;
        document.getElementById('checkoutButton').addEventListener('click', () => {
          cartContent.style.display = 'none';
          checkoutDone.classList.add('show');
          document.getElementById('orderNumber').textContent = `J-${Math.floor(100 + Math.random() * 900)}`;
        });
      }
      function openCart() { renderCart(); cartModal.classList.add('open'); }
      function closeCart() {
        cartModal.classList.remove('open');
        setTimeout(() => { cartContent.style.display = ''; checkoutDone.classList.remove('show'); }, 260);
      }
      document.getElementById('cartButton').addEventListener('click', openCart);
      document.getElementById('closeCart').addEventListener('click', closeCart);
      cartModal.addEventListener('click', event => { if (event.target === cartModal) closeCart(); });
      document.addEventListener('keydown', event => { if (event.key === 'Escape') closeCart(); });

      document.getElementById('themeToggle').addEventListener('click', () => {
        const dark = body.classList.toggle('dark');
        document.getElementById('themeToggle').setAttribute('aria-label', dark ? 'Switch to light mode' : 'Switch to dark mode');
        document.getElementById('themeIcon').innerHTML = dark
          ? '<circle cx="12" cy="12" r="4"/><path d="M12 2v2M12 20v2M4.93 4.93l1.42 1.42M17.66 17.66l1.41 1.41M2 12h2M20 12h2M4.93 19.07l1.42-1.42M17.66 6.34l1.41-1.41"/>'
          : '<path d="M20.4 14.6A8.5 8.5 0 0 1 9.4 3.6 8.5 8.5 0 1 0 20.4 14.6Z"/>';
      });

      updatePrice();
    })();
  </script>
