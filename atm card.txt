    <script>
        const cardWrapper = document.getElementById('card-wrapper');
        const card = document.getElementById('card');
        const cardFront = document.getElementById('card-front');
        const payBtn = document.getElementById('pay-btn');
        const successPopup = document.getElementById('success-popup');
        const resetBtn = document.getElementById('reset-btn');
        const gyroBtn = document.getElementById('enable-gyro');
        const flipBtn = document.getElementById('flip-to-cvv');

        // Inputs & Liquid Fills
        const inputNum = document.getElementById('card-number');
        const liquidNum = document.getElementById('liquid-num');

        const inputHolder = document.getElementById('card-holder');
        const liquidHolder = document.getElementById('liquid-holder');

        const inputExpiry = document.getElementById('card-expiry');
        const liquidExpiry = document.getElementById('liquid-expiry');

        const inputCvv = document.getElementById('card-cvv');
        const liquidCvv = document.getElementById('liquid-cvv');

        let isGyroActive = false;
        let isFlipped = false;
        let currentTiltX = 0;
        let currentTiltY = 0;

        // 1. Liquid Water Fill Calculation Function
        function updateLiquidFill(inputElement, liquidElement, maxLength) {
            let length = inputElement.value.length;
            let percentage = (length / maxLength) * 100;
            liquidElement.style.height = `${percentage}%`;
        }

        // 2. Input Listeners with Liquid Progress
        inputNum.addEventListener('input', (e) => {
            let val = e.target.value.replace(/\D/g, ''); 
            let formatted = val.match(/.{1,4}/g)?.join(' ') || '';
            e.target.value = formatted.substring(0, 19);
            updateLiquidFill(inputNum, liquidNum, 19);
        });

        inputHolder.addEventListener('input', (e) => {
            e.target.value = e.target.value.toUpperCase();
            updateLiquidFill(inputHolder, liquidHolder, 20);
        });

        inputExpiry.addEventListener('input', (e) => {
            let val = e.target.value.replace(/\D/g, '');
            if (val.length >= 3) {
                e.target.value = val.substring(0, 2) + '/' + val.substring(2, 4);
            } else {
                e.target.value = val;
            }
            updateLiquidFill(inputExpiry, liquidExpiry, 5);
        });

        inputCvv.addEventListener('input', (e) => {
            e.target.value = e.target.value.replace(/\D/g, '');
            updateLiquidFill(inputCvv, liquidCvv, 3);
        });

        // 3. Flip Controls
        flipBtn.addEventListener('click', () => {
            isFlipped = true;
            update3DRotation();
            setTimeout(() => inputCvv.focus(), 300);
        });

        inputCvv.addEventListener('focus', () => {
            isFlipped = true;
            update3DRotation();
        });

        inputCvv.addEventListener('blur', () => {
            isFlipped = false;
            update3DRotation();
        });

        // 4. Payment Trigger
        payBtn.addEventListener('click', () => {
            payBtn.disabled = true;
            payBtn.innerHTML = `<span>Processing...</span>`;

            cardFront.classList.add('scanning');

            setTimeout(() => {
                cardFront.classList.remove('scanning');
                cardWrapper.classList.add('flying-out');

                setTimeout(() => {
                    successPopup.classList.add('show');
                }, 800);

            }, 2000);
        });

        // Reset Form
        resetBtn.addEventListener('click', () => {
            successPopup.classList.remove('show');
            cardWrapper.classList.remove('flying-out');
            payBtn.disabled = false;
            payBtn.innerHTML = `<span>Pay $120.00 Now</span><span style="font-size: 1.1rem;">➔</span>`;
            
            // Clear inputs and liquid levels
            inputNum.value = ''; liquidNum.style.height = '0%';
            inputHolder.value = ''; liquidHolder.style.height = '0%';
            inputExpiry.value = ''; liquidExpiry.style.height = '0%';
            inputCvv.value = ''; liquidCvv.style.height = '0%';
        });

        // 5. Controlled 3D Motion (Very Subtle Angles)
        function update3DRotation() {
            let baseRotateY = isFlipped ? 180 : 0;
            card.style.transform = `rotateX(${currentTiltX}deg) rotateY(${baseRotateY + currentTiltY}deg)`;
        }

        // Gyroscope & Hover Logic (Decreased tilt factor for subtle movement)
        function handleOrientation(event) {
            let x = event.beta;  
            let y = event.gamma; 

            if (x === null || y === null) return;

            // Maximum +/- 6 degrees tilt for mobile gyro
            currentTiltX = -Math.min(Math.max(x - 45, -6), 6);
            currentTiltY = Math.min(Math.max(y, -6), 6);

            isGyroActive = true;
            update3DRotation();
        }

        async function requestGyroPermission() {
            if (typeof DeviceOrientationEvent !== 'undefined' && typeof DeviceOrientationEvent.requestPermission === 'function') {
                try {
                    const permissionState = await DeviceOrientationEvent.requestPermission();
                    if (permissionState === 'granted') {
                        window.addEventListener('deviceorientation', handleOrientation);
                        gyroBtn.style.display = 'none';
                    }
                } catch (error) { console.error(error); }
            } else if ('DeviceOrientationEvent' in window) {
                window.addEventListener('deviceorientation', handleOrientation);
                gyroBtn.style.display = 'none';
            }
        }

        gyroBtn.addEventListener('click', requestGyroPermission);
        if ('DeviceOrientationEvent' in window && typeof DeviceOrientationEvent.requestPermission !== 'function') {
            window.addEventListener('deviceorientation', handleOrientation);
            gyroBtn.style.display = 'none';
        }

        // Desktop Mouse Movement (Max 6 degrees rotation)
        window.addEventListener('mousemove', (e) => {
            if (isGyroActive) return;

            const rect = card.getBoundingClientRect();
            const centerX = rect.left + rect.width / 2;
            const centerY = rect.top + rect.height / 2;

            // Reduced multiplier from 18/20 down to 6 for subtle hover movement
            currentTiltX = ((e.clientY - centerY) / (rect.height / 2)) * -6;
            currentTiltY = ((e.clientX - centerX) / (rect.width / 2)) * 6;

            update3DRotation();
        });

        window.addEventListener('mouseleave', () => {
            if (!isGyroActive) {
                currentTiltX = 0;
                currentTiltY = 0;
                update3DRotation();
            }
        });
    </script>
