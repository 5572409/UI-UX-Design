script.js


const premiumToggle = document.getElementById('premiumToggle');
const toggleStatus = document.getElementById('toggleStatus');
const jellyKnob = document.querySelector('.jelly-knob');

premiumToggle.addEventListener('change', function() {
    if (this.checked) {
        // Change text state to ON
        toggleStatus.innerText = "ON";
        toggleStatus.className = "status-text text-on";
        
        // Stretchy bounce effect handle: thori der ke liye squeeze karna jab load hit ho
        jellyKnob.style.transform = "scaleX(1.3) scaleY(0.8)";
        setTimeout(() => {
            jellyKnob.style.transform = "scaleX(1) scaleY(1)";
        }, 250);
        
    } else {
        // Change text state to OFF
        toggleStatus.innerText = "OFF";
        toggleStatus.className = "status-text text-off";
        
        // Reverse stretchy effect logic
        jellyKnob.style.transform = "scaleX(1.3) scaleY(0.8)";
        setTimeout(() => {
            jellyKnob.style.transform = "scaleX(1) scaleY(1)";
        }, 250);
    }
});
