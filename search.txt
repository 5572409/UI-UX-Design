            <button class="search-btn" id="searchBtn">Search</button>
            
            <!-- Suggestions Menu -->
            <div class="suggestions-list" id="suggestionsList">
                <div class="suggestion-item">HTML Premium Tutorials</div>
                <div class="suggestion-item">HTML & CSS Advanced Concepts</div>
                <div class="suggestion-item) JavaScript Interactive UI</div>
            </div>
        </div>
    </div>

    <!-- Finding/Loading State -->
    <div class="loader-container" id="loaderContainer">
        <div class="loader-icon"></div>
        <div class="loader-text" id="loaderText">Finding "HTML PRE..."</div>
    </div>

    <script>
        const searchInput = document.getElementById('searchInput');
        const searchWrapper = document.getElementById('searchWrapper');
        const suggestionsList = document.getElementById('suggestionsList');
        const searchBtn = document.getElementById('searchBtn');
        const loaderContainer = document.getElementById('loaderContainer');
        const loaderText = document.getElementById('loaderText');

        // Focus par neon effect active karna
        searchInput.addEventListener('focus', () => {
            searchWrapper.classList.add('active');
            if(searchInput.value.trim() !== "") {
                suggestionsList.style.display = 'block';
            }
        });

        // Focus hatne par effect close karna (delay lagaya takay click register ho sakay)
        searchInput.addEventListener('blur', () => {
            setTimeout(() => {
                searchWrapper.classList.remove('active');
                suggestionsList.style.display = 'none';
            }, 200);
        });

        // Typing event for suggestions dropdown
        searchInput.addEventListener('input', (e) => {
            if(e.target.value.trim() !== "") {
                suggestionsList.style.display = 'block';
            } else {
                suggestionsList.style.display = 'none';
            }
        });

        // Dropdown item select karne par
        document.querySelectorAll('.suggestion-item').forEach(item => {
            item.addEventListener('click', (e) => {
                searchInput.value = e.target.innerText;
                suggestionsList.style.display = 'none';
            });
        });

        // Search action aur custom Loader animation run karna
        function performSearch() {
            const query = searchInput.value.trim() || "HTML Premium Tutorials";
            
            // Text truncation masking for loader text like "HTML PRE..."
            const truncatedQuery = query.length > 8 ? query.substring(0, 8).toUpperCase() + "..." : query.toUpperCase();
            loaderText.innerText = `Finding "${truncatedQuery}"`;
            
            // Loader show karna
            loaderContainer.style.display = 'block';
            
            // 2.5 seconds ke baad hide kar dena simulation ke liye
            setTimeout(() => {
                loaderContainer.style.display = 'none';
                alert(`Search complete for: ${query}`);
            }, 2500);
        }

        searchBtn.addEventListener('click', performSearch);
        
        // Enter key press support
        searchInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                performSearch();
            }
        });
    </script>
