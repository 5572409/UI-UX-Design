      debugShowCheckedModeBanner: false,

// ============ LOGIN SCREEN ============
class LoginScreen extends StatefulWidget {
  const LoginScreen({super.key});

  @override
  State<LoginScreen> createState() => _LoginScreenState();
}

class _LoginScreenState extends State<LoginScreen>
    with TickerProviderStateMixin {
  late AnimationController _screenAnimationController;
  late Animation<double> _fadeAnimation;
  late Animation<Offset> _slideAnimation;

  late AnimationController _ballController;
  late Animation<double> _ballRotation;
  late Animation<Offset> _ballPosition;
  late Animation<double> _ballScale;

  final TextEditingController _emailController = TextEditingController();
  final TextEditingController _passwordController = TextEditingController();
  bool _isButtonAnimating = false;
  bool _obscurePassword = true;
  bool _isBallMoving = false;

  @override
  void initState() {
    super.initState();

    _screenAnimationController = AnimationController(
      duration: const Duration(milliseconds: 1200),
      vsync: this,
    );

    _fadeAnimation = Tween<double>(begin: 0.0, end: 1.0).animate(
      CurvedAnimation(parent: _screenAnimationController, curve: Curves.easeIn),
    );

    _slideAnimation = Tween<Offset>(
      begin: const Offset(0, 0.3),
      end: Offset.zero,
    ).animate(
      CurvedAnimation(
          parent: _screenAnimationController, curve: Curves.easeOut),
    );

    _screenAnimationController.forward();

    _ballController = AnimationController(
      duration: const Duration(milliseconds: 2000),
      vsync: this,
    );

    _ballRotation = Tween<double>(begin: 0, end: 10 * pi).animate(
      CurvedAnimation(
          parent: _ballController, curve: Curves.easeInOut),
    );

    _ballScale = TweenSequence<double>([
      TweenSequenceItem(tween: Tween<double>(begin: 0.3, end: 1.2), weight: 30),
      TweenSequenceItem(tween: Tween<double>(begin: 1.2, end: 0.8), weight: 40),
      TweenSequenceItem(tween: Tween<double>(begin: 0.8, end: 0.5), weight: 30),
    ]).animate(_ballController);

    _ballPosition = TweenSequence<Offset>([
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: const Offset(-4.0, 0.5),
            end: const Offset(-2.0, -1.0),
          ),
          weight: 20),
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: const Offset(-2.0, -1.0),
            end: const Offset(0.0, -2.5),
          ),
          weight: 25),
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: const Offset(0.0, -2.5),
            end: const Offset(2.0, -1.5),
          ),
          weight: 25),
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: const Offset(2.0, -1.5),
            end: const Offset(4.5, 0.5),
          ),
          weight: 30),
    ]).animate(
      CurvedAnimation(parent: _ballController, curve: Curves.easeInOut),
    );
  }

  @override
  void dispose() {
    _screenAnimationController.dispose();
    _ballController.dispose();
    _emailController.dispose();
    _passwordController.dispose();
    super.dispose();
  }

  void _handleLoginClick() {
    if (_emailController.text.isEmpty || _passwordController.text.isEmpty) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(
          content: Text('Please fill all fields'),
          backgroundColor: Colors.red,
          duration: Duration(seconds: 2),
        ),
      );
      return;
    }

    setState(() {
      _isButtonAnimating = true;
      _isBallMoving = true;
    });

    _ballController.forward().then((_) {
      Future.delayed(const Duration(milliseconds: 500), () {
        if (mounted) {
          setState(() {
            _isButtonAnimating = false;
            _isBallMoving = false;
          });

          _ballController.reset();

          Navigator.pushReplacement(
            context,
            MaterialPageRoute(
              builder: (context) => const ProfileScreen(),
            ),
          );
        }
      });
    });
  }

  Widget _buildCricketBall(double size) {
    return AnimatedBuilder(
      animation: _ballController,
      builder: (context, child) {
        return Transform.scale(
          scale: _ballScale.value,
          child: Transform.rotate(
            angle: _ballRotation.value,
            child: Container(
              width: size,
              height: size,
              decoration: BoxDecoration(
                shape: BoxShape.circle,
                gradient: RadialGradient(
                  colors: [
                    Colors.red.shade700,
                    Colors.red.shade900,
                  ],
                  radius: 0.9,
                ),
                boxShadow: [
                  BoxShadow(
                    color: const Color(0xFF00FF00).withOpacity(0.4),
                    blurRadius: 25,
                    spreadRadius: 8,
                  ),
                ],
              ),
              child: CustomPaint(
                painter: CricketBallPainter(),
                size: Size(size, size),
              ),
            ),
          ),
        );
      },
    );
  }

// ===== CUSTOM PAINTER FOR CRICKET BALL =====
class CricketBallPainter extends CustomPainter {
  @override
  void paint(Canvas canvas, Size size) {
    final center = Offset(size.width / 2, size.height / 2);
    final radius = size.width * 0.4;

    final paint = Paint()
      ..color = Colors.white.withOpacity(0.4)
      ..strokeWidth = 1.5
      ..style = PaintingStyle.stroke;

    canvas.drawArc(
      Rect.fromCircle(center: center, radius: radius),
      -pi / 2,
      pi,
      false,
      paint,
    );

    for (int i = 0; i < 6; i++) {
      final angle = (i / 6) * 2 * pi + pi / 6;
      final x1 = center.dx + radius * 0.6 * cos(angle);
      final y1 = center.dy + radius * 0.6 * sin(angle);
      final x2 = center.dx + radius * 0.9 * cos(angle);
      final y2 = center.dy + radius * 0.9 * sin(angle);
      canvas.drawLine(
        Offset(x1, y1),
        Offset(x2, y2),
        paint,
      );
    }
  }

  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
}
