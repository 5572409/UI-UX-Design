    <script>
        const viewFrame = document.getElementById('viewFrame');
        const frameCounter = document.getElementById('frameCounter');

        function updateMatrix(mode, btn) {
            document.querySelectorAll('.btn-view').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');

            viewFrame.classList.remove('frame-md', 'frame-lg', 'frame-xl');

            if (mode === 'sm') {
                viewFrame.style.width = '380px';
                frameCounter.textContent = '380px [Base Mode]';
            } else if (mode === 'md') {
                viewFrame.style.width = '600px';
                viewFrame.classList.add('frame-md');
                frameCounter.textContent = '600px [Breakpoint: ≥480px]';
            } else if (mode === 'lg') {
                viewFrame.style.width = '840px';
                viewFrame.classList.add('frame-lg');
                frameCounter.textContent = '840px [Breakpoint: ≥768px]';
            } else if (mode === 'xl') {
                viewFrame.style.width = '100%';
                viewFrame.classList.add('frame-xl');
                frameCounter.textContent = '1024px+ [Breakpoint: Desktop Matrix]';
            }
        }
    </script>
