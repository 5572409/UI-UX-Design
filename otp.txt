Script:::


const fields = document.querySelectorAll('.otp-field');
const verifyBtn = document.getElementById('verifyBtn');
const card = document.getElementById('otpCard');
const msgText = document.getElementById('msgText');

// Auto-focus field shift logic
fields.forEach((field, index) => {
    field.addEventListener('input', (e) => {
        field.value = field.value.replace(/[^0-9]/g, '');
        if (field.value.length === 1 && index < fields.length - 1) {
            fields[index + 1].focus();
        }
    });

    field.addEventListener('keydown', (e) => {
        if (e.key === 'Backspace' && field.value.length === 0 && index > 0) {
            fields[index - 1].focus();
        }
    });
});

// Custom Neon Spaced Typewriter (Exact 2 Seconds Duration)
function typeNeonMessage(text, className, callback) {
    msgText.innerText = '';
    msgText.className = `message ${className}`;
    
    let currentIdx = 0;
    const speed = 2000 / text.length; // 2000ms distributed evenly

    function typing() {
        if (currentIdx < text.length) {
            msgText.innerText += text.charAt(currentIdx);
            currentIdx++;
            setTimeout(typing, speed);
        } else if (callback) {
            setTimeout(callback, 600); 
        }
    }
    typing();
}

// Verification Process
function verifyOTP() {
    let enteredOTP = '';
    fields.forEach(field => enteredOTP += field.value);

    // Style clean up
    card.classList.remove('dangerous-glitch', 'success-glow');
    document.body.classList.remove('danger-alert');
    msgText.innerText = '';

    if (enteredOTP === '1234') {
        // --- SUCCESS STATE (Short Words + Extra Space Format) ---
        card.classList.add('success-glow');
        
        // Short text with internal spacing
        typeNeonMessage("OK  PASS", "success", () => {
            fields.forEach(field => field.value = ''); // Clear numbers
            fields[0].focus();
        });

    } else {
        // --- DANGEROUS GLITCH STATE (Short Spaced) ---
        document.body.classList.add('danger-alert');
        void card.offsetWidth; 
        card.classList.add('dangerous-glitch');

        // Short dangerous text with spacing
        typeNeonMessage("WRONG  CODE", "danger", () => {
            document.body.classList.remove('danger-alert');
        });

        if (navigator.vibrate) {
            navigator.vibrate([100, 50, 100, 50, 200]); 
        }
    }
}

// Button and key interactions
verifyBtn.addEventListener('click', verifyOTP);

document.addEventListener('keydown', (e) => {
    if (e.key === 'Enter') {
        verifyOTP();
    }
});
