    <script>
        // Core Event Engine
        function setupStandardPagination(containerSelector) {
            const container = document.querySelector(containerSelector);
            const items = container.querySelectorAll('.page-item');
            const prev = container.querySelector('.prev');
            const next = container.querySelector('.next');

            items.forEach(item => {
                item.addEventListener('click', () => {
                    container.querySelector('.page-item.active').classList.remove('active');
                    item.classList.add('active');
                });
            });

            prev.addEventListener('click', () => {
                const active = container.querySelector('.page-item.active');
                const prevElem = active.previousElementSibling;
                if (prevElem && prevElem.classList.contains('page-item')) {
                    active.classList.remove('active');
                    prevElem.classList.add('active');
                }
            });

            next.addEventListener('click', () => {
                const active = container.querySelector('.page-item.active');
                const nextElem = active.nextElementSibling;
                if (nextElem && nextElem.classList.contains('page-item')) {
                    active.classList.remove('active');
                    nextElem.classList.add('active');
                }
            });
        }

        setupStandardPagination('.classic');
        setupStandardPagination('.pills');
        setupStandardPagination('.dots-style');
        setupStandardPagination('.bordered');

        // 5. Track Logic
        let trackPage = 3;
        const totalTrackPages = 5;
        const trackNodes = document.querySelectorAll('.track-node');
        const trackProgress = document.getElementById('track-progress');
        
        function updateTrack(page) {
            trackPage = Math.max(1, Math.min(page, totalTrackPages));
            trackNodes.forEach((node, idx) => {
                if (idx + 1 === trackPage) node.classList.add('active');
                else node.classList.remove('active');
            });
            const progressPercent = ((trackPage - 1) / (totalTrackPages - 1)) * 100;
            trackProgress.style.width = `${progressPercent}%`;
        }
        document.getElementById('track-prev').addEventListener('click', () => updateTrack(trackPage - 1));
        document.getElementById('track-next').addEventListener('click', () => updateTrack(trackPage + 1));
        trackNodes.forEach(node => {
            node.addEventListener('click', (e) => updateTrack(parseInt(e.target.dataset.page)));
        });

        // 6. Compact Logic
        let compactPage = 3;
        const compVal = document.getElementById('comp-val');
        document.getElementById('comp-prev').addEventListener('click', () => {
            if (compactPage > 1) { compactPage--; compVal.innerText = compactPage; }
        });
        document.getElementById('comp-next').addEventListener('click', () => {
            if (compactPage < 5) { compactPage++; compVal.innerText = compactPage; }
        });

        // 7. Progress Logic
        let progressPage = 3;
        const progFill = document.getElementById('prog-fill');
        const progVal = document.getElementById('prog-val');
        function updateProgressBar() {
            progVal.innerText = progressPage;
            progFill.style.width = `${(progressPage / 5) * 100}%`;
        }
        document.getElementById('prog-prev').addEventListener('click', () => {
            if (progressPage > 1) { progressPage--; updateProgressBar(); }
        });
        document.getElementById('prog-next').addEventListener('click', () => {
            if (progressPage < 5) { progressPage++; updateProgressBar(); }
        });

        // 8. Steps Logic
        let stepPage = 3;
        const stepDots = document.querySelectorAll('.step-dot');
        function updateSteps() {
            stepDots.forEach((dot, idx) => {
                const currentStep = idx + 1;
                dot.className = 'step-dot';
                if (currentStep < stepPage) dot.classList.add('completed');
                else if (currentStep === stepPage) dot.classList.add('active');
            });
        }
        document.getElementById('step-prev').addEventListener('click', () => {
            if (stepPage > 1) { stepPage--; updateSteps(); }
        });
        document.getElementById('step-next').addEventListener('click', () => {
            if (stepPage < 5) { stepPage++; updateSteps(); }
        });
        stepDots.forEach(dot => {
            dot.addEventListener('click', (e) => {
                stepPage = parseInt(e.target.dataset.step);
                updateSteps();
            });
        });
    </script>
