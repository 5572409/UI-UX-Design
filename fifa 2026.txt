// ============ FIFA WORLD CUP 2026 LOGIN SCREEN ============
class LoginScreen extends StatefulWidget {
  const LoginScreen({super.key});

  @override
  State<LoginScreen> createState() => _LoginScreenState();
}

class _LoginScreenState extends State<LoginScreen>
    with TickerProviderStateMixin {
  // Screen entrance animations
  late AnimationController _screenAnimationController;
  late Animation<double> _fadeAnimation;
  late Animation<Offset> _slideAnimation;

  // Football animation
  late AnimationController _footballController;
  late Animation<double> _footballRotation;
  late Animation<Offset> _footballPosition;
  late Animation<double> _footballScale;

  // Celebration particles
  late AnimationController _celebrationController;
  List<CelebrationParticle> _particles = [];
  final Random _random = Random();

  // Confetti animation
  late AnimationController _confettiController;
  List<Confetti> _confetti = [];

  final TextEditingController _emailController = TextEditingController();
  final TextEditingController _passwordController = TextEditingController();
  bool _isButtonAnimating = false;
  bool _obscurePassword = true;
  bool _showCelebration = false;
  bool _isFootballMoving = false;

  @override
  void initState() {
    super.initState();

    // ===== SCREEN ENTRANCE ANIMATION =====
    _screenAnimationController = AnimationController(
      duration: const Duration(milliseconds: 1200),
      vsync: this,
    );

    _fadeAnimation = Tween<double>(begin: 0.0, end: 1.0).animate(
      CurvedAnimation(parent: _screenAnimationController, curve: Curves.easeIn),
    );

    _slideAnimation = Tween<Offset>(
      begin: const Offset(0, 0.3),
      end: Offset.zero,
    ).animate(
      CurvedAnimation(
          parent: _screenAnimationController, curve: Curves.easeOut),
    );

    _screenAnimationController.forward();

    // ===== FOOTBALL ANIMATION =====
    _footballController = AnimationController(
      duration: const Duration(milliseconds: 3000),
      vsync: this,
    );

    _footballRotation = Tween<double>(begin: 0, end: 8 * pi).animate(
      CurvedAnimation(
          parent: _footballController, curve: Curves.easeInOut),
    );

    _footballScale = TweenSequence<double>([
      TweenSequenceItem(tween: Tween<double>(begin: 0.5, end: 1.5), weight: 25),
      TweenSequenceItem(tween: Tween<double>(begin: 1.5, end: 0.8), weight: 25),
      TweenSequenceItem(tween: Tween<double>(begin: 0.8, end: 1.2), weight: 25),
      TweenSequenceItem(tween: Tween<double>(begin: 1.2, end: 1.0), weight: 25),
    ]).animate(_footballController);

    _footballPosition = TweenSequence<Offset>([
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: Offset.zero,
            end: const Offset(-3.0, -4.0),
          ),
          weight: 20),
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: const Offset(-3.0, -4.0),
            end: const Offset(3.0, -5.0),
          ),
          weight: 20),
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: const Offset(3.0, -5.0),
            end: const Offset(-2.0, -2.0),
          ),
          weight: 20),
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: const Offset(-2.0, -2.0),
            end: const Offset(2.0, -3.0),
          ),
          weight: 20),
      TweenSequenceItem(
          tween: Tween<Offset>(
            begin: const Offset(2.0, -3.0),
            end: Offset.zero,
          ),
          weight: 20),
    ]).animate(
      CurvedAnimation(parent: _footballController, curve: Curves.easeInOut),
    );

    // ===== CELEBRATION PARTICLES =====
    _celebrationController = AnimationController(
      duration: const Duration(milliseconds: 3000),
      vsync: this,
    )..addListener(() {
      if (_showCelebration) {
        setState(() {
          // Update particles
          for (var particle in _particles) {
            particle.x += particle.vx;
            particle.y += particle.vy;
            particle.vy += 0.1; // Gravity
            particle.life -= 0.005;
            particle.size *= 0.995;
          }
          _particles.removeWhere((p) => p.life <= 0 || p.size < 1);
        });
      }
    });

    // ===== CONFETTI ANIMATION =====
    _confettiController = AnimationController(
      duration: const Duration(milliseconds: 4000),
      vsync: this,
    )..addListener(() {
      if (_showCelebration) {
        setState(() {
          for (var confetti in _confetti) {
            confetti.x += confetti.vx;
            confetti.y += confetti.vy;
            confetti.vy += 0.05;
            confetti.rotation += 0.02;
          }
        });
      }
    });
  }

  void _generateCelebration() {
    final size = MediaQuery.of(context).size;
    final centerX = size.width / 2;
    final centerY = size.height / 2;

    // Generate particles in FIFA colors
    final colors = [
      const Color(0xFFFFD700), // Gold
      const Color(0xFFDAA520), // Goldenrod
      Colors.white,
      Colors.blue,
      Colors.red,
      Colors.green,
      Colors.purple,
      Colors.orange,
    ];

    for (int i = 0; i < 50; i++) {
      final angle = _random.nextDouble() * 2 * pi;
      final speed = 5 + _random.nextDouble() * 10;
      _particles.add(
        CelebrationParticle(
          x: centerX,
          y: centerY,
          vx: cos(angle) * speed,
          vy: sin(angle) * speed - 5,
          life: 0.8 + _random.nextDouble() * 0.2,
          size: 5 + _random.nextDouble() * 10,
          color: colors[_random.nextInt(colors.length)],
        ),
      );
    }

    // Generate confetti
    for (int i = 0; i < 30; i++) {
      _confetti.add(
        Confetti(
          x: _random.nextDouble() * size.width,
          y: _random.nextDouble() * size.height * 0.5,
          vx: -2 + _random.nextDouble() * 4,
          vy: 2 + _random.nextDouble() * 4,
          rotation: _random.nextDouble() * 2 * pi,
          size: 8 + _random.nextDouble() * 12,
          color: colors[_random.nextInt(colors.length)],
        ),
      );
    }
  }

  @override
  void dispose() {
    _screenAnimationController.dispose();
    _footballController.dispose();
    _celebrationController.dispose();
    _confettiController.dispose();
    _emailController.dispose();
    _passwordController.dispose();
    super.dispose();
  }

  void _handleLoginClick() {
    if (_emailController.text.isEmpty || _passwordController.text.isEmpty) {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(
          content: Text('Please fill all fields'),
          backgroundColor: Colors.red,
          duration: Duration(seconds: 2),
        ),
      );
      return;
    }

    setState(() {
      _isButtonAnimating = true;
      _isFootballMoving = true;
    });

    // Start football animation
    _footballController.forward().then((_) {
      setState(() {
        _showCelebration = true;
        _isFootballMoving = false;
      });

      _generateCelebration();
      _celebrationController.repeat();
      _confettiController.repeat();

      // After celebration, show success
      Future.delayed(const Duration(seconds: 4), () {
        if (mounted) {
          setState(() {
            _isButtonAnimating = false;
            _showCelebration = false;
            _particles.clear();
            _confetti.clear();
          });

          ScaffoldMessenger.of(context).showSnackBar(
            const SnackBar(
              content: Text('⚽ Goal! Login Successful! 🎉'),
              backgroundColor: Colors.green,
              duration: Duration(seconds: 3),
            ),
          );

          _footballController.reset();
          _celebrationController.stop();
          _confettiController.stop();
        }
      });
    });
  }

    return AnimatedBuilder(

        return Transform.scale(
          scale: _footballScale.value,

                  radius: 0.9,
                ),
                boxShadow: [
                  BoxShadow(
                    color: const Color(0xFFFFD700).withOpacity(0.6),
                    blurRadius: 30,
                    spreadRadius: 10,
                  ),
                ],
              ),
              child: CustomPaint(
                painter: FootballPainter(),
                size: Size(size, size),
              ),
            ),
          ),
        );
      },
    );
  }

// ===== CUSTOM PAINTER FOR FOOTBALL =====
class FootballPainter extends CustomPainter {
  @override
  void paint(Canvas canvas, Size size) {
    final center = Offset(size.width / 2, size.height / 2);
    final radius = size.width * 0.35;

    // Draw pentagon shapes
    final paint = Paint()
      ..color = Colors.black.withOpacity(0.3)
      ..style = PaintingStyle.fill;

    // Draw 5 pentagons around the ball
    for (int i = 0; i < 5; i++) {
      final angle = (i / 5) * 2 * pi - pi / 2;
      final x = center.dx + radius * cos(angle);
      final y = center.dy + radius * sin(angle);
      final pentagonSize = size.width * 0.12;

      // Draw pentagon using path
      final path = Path();
      for (int j = 0; j < 5; j++) {
        final pentAngle = (j / 5) * 2 * pi - pi / 2;
        final px = x + pentagonSize * cos(pentAngle);
        final py = y + pentagonSize * sin(pentAngle);
        if (j == 0) {
          path.moveTo(px, py);
        } else {
          path.lineTo(px, py);
        }
      }
      path.close();
      canvas.drawPath(path, paint);
    }

    // Draw center pentagon
    final centerPath = Path();
    final centerPentSize = size.width * 0.12;
    for (int j = 0; j < 5; j++) {
      final pentAngle = (j / 5) * 2 * pi - pi / 2;
      final px = center.dx + centerPentSize * cos(pentAngle);
      final py = center.dy + centerPentSize * sin(pentAngle);
      if (j == 0) {
        centerPath.moveTo(px, py);
      } else {
        centerPath.lineTo(px, py);
      }
    }
    centerPath.close();
    canvas.drawPath(centerPath, paint);

    // Draw connecting lines
    final linePaint = Paint()
      ..color = Colors.black.withOpacity(0.2)
      ..strokeWidth = 1.5
      ..style = PaintingStyle.stroke;

    for (int i = 0; i < 5; i++) {
      final angle1 = (i / 5) * 2 * pi - pi / 2;
      final angle2 = ((i + 1) / 5) * 2 * pi - pi / 2;
      final p1 = Offset(
        center.dx + radius * cos(angle1),
        center.dy + radius * sin(angle1),
      );
      final p2 = Offset(
        center.dx + radius * cos(angle2),
        center.dy + radius * sin(angle2),
      );
      canvas.drawLine(p1, p2, linePaint);
    }
  }

  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
}

// ===== CELEBRATION PARTICLE CLASS =====
class CelebrationParticle {
  double x;
  double y;
  double vx;
  double vy;
  double life;
  double size;
  Color color;
